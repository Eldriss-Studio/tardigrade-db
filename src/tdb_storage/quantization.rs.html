1//! Quantization strategies for KV cache tensors.
2//!
3//! Implements the Strategy pattern: `QuantizeStrategy` and `DequantizeStrategy` traits.
4//! Supported formats: Q4 (group-wise 4-bit), Q8 (symmetric INT8).
5
6/// Group size for block quantization. Each group shares one scale factor.
7const Q4_GROUP_SIZE: usize = 32;
8
9/// Opaque container for quantized tensor data.
10/// Includes the quantized bytes plus per-group scale factors needed for dequantization.
11#[derive(Debug, Clone)]
12pub struct QuantizedTensor {
13    /// Packed quantized values. For Q4: two 4-bit values per byte.
14    pub data: Vec<u8>,
15    /// One f32 scale factor per group.
16    pub scales: Vec<f32>,
17    /// Number of original f32 elements.
18    pub original_len: usize,
19}
20
21/// Quantize f32 vectors into a compressed representation.
22pub trait QuantizeStrategy {
23    fn quantize(values: &[f32]) -> QuantizedTensor;
24}
25
26/// Dequantize compressed data back to f32 vectors.
27pub trait DequantizeStrategy {
28    fn dequantize(tensor: &QuantizedTensor) -> Vec<f32>;
29}
30
31/// Group-wise 4-bit quantization (`Q4_0`).
32///
33/// For each group of 32 values:
34/// 1. Find `abs_max` — the maximum absolute value in the group.
35/// 2. Compute `scale = abs_max / 7.0` (mapping to signed range [-7, 7]).
36/// 3. Quantize each value: `q = round(value / scale) + 8` clamped to [0, 15].
37/// 4. Pack two 4-bit values per byte (low nibble first).
38///
39/// This matches the GGML `Q4_0` scheme used in llama.cpp.
40#[derive(Debug)]
41pub struct Q4;
42
43impl QuantizeStrategy for Q4 {
44    fn quantize(values: &[f32]) -> QuantizedTensor {
45        let num_groups = values.len().div_ceil(Q4_GROUP_SIZE);
46        let packed_bytes_per_group = Q4_GROUP_SIZE / 2; // 2 values per byte
47        let mut data = Vec::with_capacity(num_groups * packed_bytes_per_group);
48        let mut scales = Vec::with_capacity(num_groups);
49
50        for group in values.chunks(Q4_GROUP_SIZE) {
51            let abs_max = group.iter().map(|v| v.abs()).fold(0.0f32, f32::max);
52
53            let scale = if abs_max == 0.0 { 1.0 } else { abs_max / 7.0 };
54            scales.push(scale);
55
56            // Quantize and pack two values per byte.
57            let mut i = 0;
58            while i < group.len() {
59                let q_low = quantize_value(group[i], scale);
60                let q_high = if i + 1 < group.len() {
61                    quantize_value(group[i + 1], scale)
62                } else {
63                    8 // zero-point for padding
64                };
65                data.push((q_high << 4) | q_low);
66                i += 2;
67            }
68
69            // Pad if group was smaller than Q4_GROUP_SIZE (last group).
70            let packed_count = group.len().div_ceil(2);
71            let pad_count = packed_bytes_per_group - packed_count;
72            data.extend(std::iter::repeat_n(0x88u8, pad_count));
73        }
74
75        QuantizedTensor { data, scales, original_len: values.len() }
76    }
77}

79impl DequantizeStrategy for Q4 {
80    fn dequantize(tensor: &QuantizedTensor) -> Vec<f32> {
81        let mut result = Vec::with_capacity(tensor.original_len);
82        let packed_bytes_per_group = Q4_GROUP_SIZE / 2;
83
84        for (group_idx, &scale) in tensor.scales.iter().enumerate() {
85            let byte_offset = group_idx * packed_bytes_per_group;
86            let group_start = group_idx * Q4_GROUP_SIZE;
87
88            for j in 0..packed_bytes_per_group {
89                if group_start + j * 2 >= tensor.original_len {
90                    break;
91                }
92                let byte = tensor.data[byte_offset + j];
93                let q_low = i16::from(byte & 0x0F) - 8;
94                result.push(q_low as f32 * scale);
95
96                if group_start + j * 2 + 1 >= tensor.original_len {
97                    break;
98                }
99                let q_high = i16::from((byte >> 4) & 0x0F) - 8;
100                result.push(q_high as f32 * scale);
101            }
102        }
103
104        result
105    }
106}

108/// Map a single f32 value to a 4-bit unsigned integer [0, 15] with zero-point at 8.
109#[inline]
110fn quantize_value(value: f32, scale: f32) -> u8 {
111    let q = (value / scale).round() as i8;
112    (q + 8).clamp(0, 15) as u8
113}

115#[cfg(test)]
116mod tests {
117    use super::*;
118
119    #[test]
120    fn test_q4_round_trip_zeros() {
121        let values = vec![0.0; 64];
122        let quantized = Q4::quantize(&values);
123        let restored = Q4::dequantize(&quantized);
124        assert_eq!(restored.len(), values.len());
125        for v in &restored {
126            assert!(v.abs() < f32::EPSILON, "Expected 0.0, got {v}");
127        }
128    }
129
130    #[test]
131    fn test_q4_round_trip_preserves_length() {
132        for len in [1, 15, 31, 32, 33, 63, 64, 100, 128, 255, 256] {
133            let values: Vec<f32> = (0..len).map(|i| (i as f32) * 0.1).collect();
134            let quantized = Q4::quantize(&values);
135            let restored = Q4::dequantize(&quantized);
136            assert_eq!(restored.len(), values.len(), "Length mismatch for input size {len}");
137        }
138    }
139
140    #[test]
141    fn test_q4_mse_within_tolerance() {
142        let values: Vec<f32> = (0..256).map(|i| ((i as f32) * 0.1).sin()).collect();
143        let quantized = Q4::quantize(&values);
144        let restored = Q4::dequantize(&quantized);
145
146        let mse: f32 =
147            values.iter().zip(restored.iter()).map(|(a, b)| (a - b) * (a - b)).sum::<f32>()
148                / values.len() as f32;
149
150        assert!(mse < 0.01, "MSE {mse:.6} exceeds 0.01");
151    }
152
153    #[test]
154    fn test_q4_compression_ratio() {
155        let values = vec![1.0f32; 128];
156        let quantized = Q4::quantize(&values);
157        let original_bytes = values.len() * 4; // f32 = 4 bytes
158        let compressed_bytes = quantized.data.len() + quantized.scales.len() * 4;
159        let ratio = original_bytes as f32 / compressed_bytes as f32;
160        // Q4 should achieve ~4x compression (4 bits vs 32 bits, plus scale overhead).
161        assert!(ratio > 3.0, "Compression ratio {ratio:.1}x is below 3x minimum");
162    }
