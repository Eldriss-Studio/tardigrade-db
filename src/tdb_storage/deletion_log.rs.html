1//! Append-only deletion log — durable persistence for pack deletions.
2//!
3//! Records `PackId` values that have been deleted. On open, the file is
4//! replayed into a `HashSet` which serves as a negative filter over the
5//! pack directory.
6//!
7//! # File format
8//!
9//! ```text
10//! ┌────────────────────────────┐
11//! │ pack_id: u64 (8 bytes, LE) │
12//! ├────────────────────────────┤
13//! │ … next record …            │
14//! └────────────────────────────┘
15//! ```
16//!
17//! Each record is exactly 8 bytes. Truncated trailing bytes (< 8) are
18//! silently discarded.
19//!
20//! # Durability contract
21//!
22//! [`DeletionLog::mark_deleted`] appends and fsyncs before returning.
23//! A crash after `mark_deleted` returns guarantees the deletion is on disk.
24//!
25//! # Recovery contract
26//!
27//! [`DeletionLog::open`] rebuilds the `HashSet` from the file. On engine
28//! open, the deleted set is applied as a filter to the pack directory.
29
30use std::collections::HashSet;
31use std::fs::{File, OpenOptions};
32use std::io::{self, Read, Write};
33use std::path::{Path, PathBuf};
34
35use tdb_core::kv_pack::PackId;
36
37const DELETION_LOG_FILENAME: &str = "deletion_log.bin";
38const RECORD_SIZE: usize = 8; // u64 pack_id
39
40/// Append-only log of deleted pack IDs.
41#[derive(Debug)]
42pub struct DeletionLog {
43    path: PathBuf,
44    deleted: HashSet<PackId>,
45}
46
47impl DeletionLog {
48    /// Open or create the deletion log in the given directory.
49    ///
50    /// Scans the file to rebuild the deleted set. Truncated trailing
51    /// bytes are silently discarded.
52    pub fn open(dir: &Path) -> io::Result<Self> {
53        let path = dir.join(DELETION_LOG_FILENAME);
54        let deleted = if path.exists() { Self::replay(&path)? } else { HashSet::new() };
55        Ok(Self { path, deleted })
56    }
57
58    /// Re-read the on-disk file and rebuild the deleted set.
59    ///
60    /// Used by `Engine::refresh` (in `tdb-engine`) to pick up deletions performed by another
61    /// `Engine` handle at the same path. Idempotent: repeated calls with no
62    /// on-disk changes leave the set unchanged.
63    pub fn refresh(&mut self) -> io::Result<()> {
64        self.deleted = if self.path.exists() { Self::replay(&self.path)? } else { HashSet::new() };
65        Ok(())
66    }
67
68    /// Mark a pack as deleted. Appends to the file and fsyncs.
69    pub fn mark_deleted(&mut self, pack_id: PackId) -> io::Result<()> {
70        let mut file = OpenOptions::new().create(true).append(true).open(&self.path)?;
71        file.write_all(&pack_id.to_le_bytes())?;
72        file.sync_all()?;
73        self.deleted.insert(pack_id);
74        Ok(())
75    }
76
77    /// Check if a pack has been deleted.
78    pub fn is_deleted(&self, pack_id: PackId) -> bool {
79        self.deleted.contains(&pack_id)
80    }
81
82    /// The complete set of deleted pack IDs.
83    pub fn deleted_set(&self) -> &HashSet<PackId> {
84        &self.deleted
85    }
86
87    /// Replay the deletion log file, rebuilding the deleted set.
88    fn replay(path: &Path) -> io::Result<HashSet<PackId>> {
89        let file_len = std::fs::metadata(path)?.len() as usize;
90        let mut data = vec![0u8; file_len];
91        let mut file = File::open(path)?;
92        file.read_exact(&mut data)?;
93
94        let mut deleted = HashSet::new();
95        let mut cursor = 0;
96
97        while cursor + RECORD_SIZE <= data.len() {
98            let bytes: [u8; 8] = data[cursor..cursor + RECORD_SIZE]
99                .try_into()
100                .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "truncated pack_id"))?;
101            let pack_id = u64::from_le_bytes(bytes);
102            deleted.insert(pack_id);
103            cursor += RECORD_SIZE;
104        }
105
106        Ok(deleted)
107    }
108}

110#[cfg(test)]
111mod tests {
112    use super::*;
113
114    #[test]
115    fn test_deletion_log_round_trip() {
116        let dir = tempfile::tempdir().unwrap();
117        let mut log = DeletionLog::open(dir.path()).unwrap();
118
119        log.mark_deleted(1).unwrap();
120        log.mark_deleted(5).unwrap();
121
122        assert!(log.is_deleted(1));
123        assert!(log.is_deleted(5));
124        assert!(!log.is_deleted(3));
125    }
126
127    #[test]
128    fn test_deletion_log_survives_reopen() {
129        let dir = tempfile::tempdir().unwrap();
130
131        {
132            let mut log = DeletionLog::open(dir.path()).unwrap();
133            log.mark_deleted(42).unwrap();
134        }
135
136        let log = DeletionLog::open(dir.path()).unwrap();
137        assert!(log.is_deleted(42));
138    }
139
140    #[test]
141    fn test_deletion_log_opens_fresh_directory() {
142        let dir = tempfile::tempdir().unwrap();
143        let log = DeletionLog::open(dir.path()).unwrap();
144        assert!(log.deleted_set().is_empty());
145    }
146
147    #[test]
148    fn test_deletion_log_duplicate_is_idempotent() {
149        let dir = tempfile::tempdir().unwrap();
150        let mut log = DeletionLog::open(dir.path()).unwrap();
151
152        log.mark_deleted(7).unwrap();
153        log.mark_deleted(7).unwrap();
154
155        assert!(log.is_deleted(7));
156        assert_eq!(log.deleted_set().len(), 1);
157    }
158
159    #[test]
160    fn test_deletion_log_truncated_trailing_bytes_discarded() {
161        let dir = tempfile::tempdir().unwrap();
162        let path = dir.path().join(DELETION_LOG_FILENAME);
163
164        // Write a valid record.
165        {
166            let mut log = DeletionLog::open(dir.path()).unwrap();
167            log.mark_deleted(1).unwrap();
168        }
169
170        // Append 3 garbage bytes (< 8, so incomplete record).
171        {
172            let mut file = OpenOptions::new().append(true).open(&path).unwrap();
173            file.write_all(&[0xFF, 0xFF, 0xFF]).unwrap();
174            file.sync_all().unwrap();
175        }
176
177        let log = DeletionLog::open(dir.path()).unwrap();
178        assert!(log.is_deleted(1));
179        assert_eq!(log.deleted_set().len(), 1);
180    }
