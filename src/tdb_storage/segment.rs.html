1//! Segment file management — append-only binary files containing serialized memory cells.
2//!
3//! Each segment file has a small header followed by variable-length records.
4//! New segments are created when the current segment exceeds the size threshold.
5//!
6//! ## Record Layout (binary, little-endian)
7//!
8//! ```text
9//! [record_len: u32]       — total bytes of this record (excluding this field)
10//! [cell_id: u64]          — globally unique cell ID
11//! [owner: u64]            — owner/agent ID
12//! [layer: u16]            — transformer layer index
13//! [key_dim: u32]          — number of f32 elements in key vector
14//! [value_dim: u32]        — number of f32 elements in value vector
15//! [pos_dim: u32]          — number of f32 elements in pos_encoding
16//! [token_span_start: u64] — token span start
17//! [token_span_end: u64]   — token span end
18//! [created_at: u64]       — creation timestamp (nanos)
19//! [updated_at: u64]       — last update timestamp (nanos)
20//! [importance: f32]       — importance score
21//! [tags: u32]             — tag bitfield
22//! [tier: u8]              — maturity tier enum
23//! [key_scales_len: u32]   — number of Q4 scale floats for key
24//! [key_data_len: u32]     — number of Q4 packed bytes for key
25//! [val_scales_len: u32]   — number of Q4 scale floats for value
26//! [val_data_len: u32]     — number of Q4 packed bytes for value
27//! [key_scales: ...]       — f32 scale factors
28//! [key_data: ...]         — packed Q4 bytes
29//! [val_scales: ...]       — f32 scale factors
30//! [val_data: ...]         — packed Q4 bytes
31//! [pos_encoding: ...]     — f32 values (unquantized)
32//! ```

44/// File header magic bytes to identify segment files.
45const SEGMENT_MAGIC: &[u8; 4] = b"TDBS";
46/// Current segment format version.
47const SEGMENT_VERSION: u32 = 1;
48/// Size of the file header: magic(4) + version(4) = 8 bytes.
49const FILE_HEADER_SIZE: u64 = 8;

58/// A single segment file — append-only storage for serialized memory cells.
59#[derive(Debug)]
60pub struct Segment {
61    id: u32,
62    path: PathBuf,
63    size: u64,
64}

66impl Segment {
67    /// Create a new empty segment file.
68    pub fn create(dir: &Path, id: u32) -> io::Result<Self> {
69        let path = segment_path(dir, id);
70        let mut file = File::create(&path)?;
71        file.write_all(SEGMENT_MAGIC)?;
72        file.write_all(&SEGMENT_VERSION.to_le_bytes())?;
73        file.flush()?;
74        Ok(Self { id, path, size: FILE_HEADER_SIZE })
75    }
76
77    /// Open an existing segment file.
78    pub fn open(dir: &Path, id: u32) -> io::Result<Self> {
79        let path = segment_path(dir, id);
80        let meta = fs::metadata(&path)?;
81        Ok(Self { id, path, size: meta.len() })
82    }

85        self.id

92    /// Append a `MemoryCell` (quantized to Q4) and return its byte offset.
93    pub fn append(&mut self, cell: &MemoryCell) -> io::Result<u64> {
94        let file = OpenOptions::new().append(true).open(&self.path)?;
95        let mut w = BufWriter::new(file);
96
97        let offset = self.size;
98        let record_bytes = write_cell_record(&mut w, cell)?;
99        self.size = offset + 4 + record_bytes as u64;
100
101        let inner = w.into_inner().map_err(std::io::IntoInnerError::into_error)?;
102        inner.sync_data()?;
103        Ok(offset)
104    }
105
106    /// Append multiple cells in a single write + single fsync (Write-Behind Buffer).
107    ///
108    /// Returns a `Vec` of (`byte_offset`, `record_size`) for each cell written.
109    /// All cells are durably committed after the single `sync_data()` call.
110    pub fn append_batch(&mut self, cells: &[MemoryCell]) -> io::Result<Vec<u64>> {
111        if cells.is_empty() {
112            return Ok(Vec::new());
113        }
114
115        let file = OpenOptions::new().append(true).open(&self.path)?;
116        let mut w = BufWriter::new(file);
117
118        let mut offsets = Vec::with_capacity(cells.len());
119        for cell in cells {
120            let offset = self.size;
121            let record_bytes = write_cell_record(&mut w, cell)?;
122            self.size = offset + 4 + record_bytes as u64;
123            offsets.push(offset);
124        }
125
126        // Single fsync for all cells — this is the key optimization.
127        let inner = w.into_inner().map_err(std::io::IntoInnerError::into_error)?;
128        inner.sync_data()?;
129        Ok(offsets)
130    }
131
132    /// Read a `MemoryCell` from a specific byte offset.
133    pub fn read_at(&self, byte_offset: u64) -> io::Result<MemoryCell> {
134        let mut file = File::open(&self.path)?;
135        file.seek(SeekFrom::Start(byte_offset))?;
136
137        let record_len = read_u32(&mut file)?;
138        let _ = record_len; // We read field-by-field, not by total length.
139
140        let cell_id: CellId = read_u64(&mut file)?;
141        let owner: OwnerId = read_u64(&mut file)?;
142        let layer: LayerId = read_u16(&mut file)?;
143        let key_dim = read_u32(&mut file)? as usize;
144        let value_dim = read_u32(&mut file)? as usize;
145        let pos_dim = read_u32(&mut file)? as usize;
146        let token_span_start = read_u64(&mut file)?;
147        let token_span_end = read_u64(&mut file)?;
148        let created_at = read_u64(&mut file)?;
149        let updated_at = read_u64(&mut file)?;
150        let importance = read_f32(&mut file)?;
151        let tags = read_u32(&mut file)?;
152        let tier_byte = read_u8(&mut file)?;
153
154        let key_scales_len = read_u32(&mut file)? as usize;
155        let key_data_len = read_u32(&mut file)? as usize;
156        let val_scales_len = read_u32(&mut file)? as usize;
157        let val_data_len = read_u32(&mut file)? as usize;
158
159        let key_scales = read_f32_vec(&mut file, key_scales_len)?;
160        let key_data = read_bytes(&mut file, key_data_len)?;
161        let val_scales = read_f32_vec(&mut file, val_scales_len)?;
162        let val_data = read_bytes(&mut file, val_data_len)?;
163        let pos_encoding = read_f32_vec(&mut file, pos_dim)?;
164
165        let key_q = QuantizedTensor { data: key_data, scales: key_scales, original_len: key_dim };
166        let val_q = QuantizedTensor { data: val_data, scales: val_scales, original_len: value_dim };
167
168        let key = Q4::dequantize(&key_q);
169        let value = Q4::dequantize(&val_q);
170
171        let tier = match tier_byte {
172            0 => Tier::Draft,
173            1 => Tier::Validated,
174            2 => Tier::Core,
175            _ => {
176                return Err(io::Error::new(
177                    io::ErrorKind::InvalidData,
178                    format!("invalid tier byte {tier_byte} in cell {cell_id}"),
179                ));
180            }
181        };
182
183        Ok(MemoryCell {
184            id: cell_id,
185            owner,
186            layer,
187            key,
188            value,
189            token_span: (token_span_start, token_span_end),
190            pos_encoding,
191            meta: CellMeta { created_at, updated_at, importance, tags, tier },
192        })
193    }
194}

196/// Scan a segment to rebuild the index (used on recovery).
197pub fn scan_segment(dir: &Path, segment_id: u32) -> io::Result<Vec<(CellId, u64)>> {
198    let path = segment_path(dir, segment_id);
199    let mut file = File::open(&path)?;
200    let file_len = file.metadata()?.len();
201
202    // Validate file header.
203    if file_len < FILE_HEADER_SIZE {
204        return Ok(Vec::new());
205    }
206    let mut magic = [0u8; 4];
207    file.read_exact(&mut magic)?;
208    if &magic != SEGMENT_MAGIC {
209        return Err(io::Error::new(
210            io::ErrorKind::InvalidData,
211            format!("invalid segment magic in {}", path.display()),
212        ));
213    }
214    // Skip version (4 bytes) — already past magic.
215    file.seek(SeekFrom::Start(FILE_HEADER_SIZE))?;
216
217    let mut entries = Vec::new();
218    let mut pos = FILE_HEADER_SIZE;
219
220    while pos < file_len {
221        let record_len = match read_u32(&mut file) {
222            Ok(len) => len as u64,
223            Err(_) => break, // truncated record_len — stop scanning
224        };
225
226        // Bounds check: ensure the full record fits within the file.
227        if pos + 4 + record_len > file_len {
228            break; // partial record at EOF — discard it
229        }
230
231        let Ok(cell_id) = read_u64(&mut file) else {
232            break; // partial record — stop scanning
233        };
234        entries.push((cell_id, pos));
235
236        // Skip the rest of this record.
237        let remaining = record_len.saturating_sub(8);
238        let Ok(seek_offset) = i64::try_from(remaining) else {
239            break;
240        };
241        if file.seek(SeekFrom::Current(seek_offset)).is_err() {
242            break;
243        }
244        pos += 4 + record_len;
245    }
246
247    Ok(entries)
248}

254/// List segment IDs found in a directory, sorted.
255pub fn list_segments(dir: &Path) -> io::Result<Vec<u32>> {
256    let mut ids = Vec::new();
257    for entry in fs::read_dir(dir)? {
258        let entry = entry?;
259        let name = entry.file_name();
260        let name = name.to_string_lossy();
261        if let Some(rest) = name.strip_prefix("segment_") {
262            if let Some(num_str) = rest.strip_suffix(".tdb") {
263                if let Ok(id) = num_str.parse::<u32>() {
264                    ids.push(id);
265                }
266            }
267        }
268    }
269    ids.sort_unstable();
270    Ok(ids)
271}

273fn compute_record_size(key_q: &QuantizedTensor, val_q: &QuantizedTensor, pos_len: usize) -> usize {
274    // Fixed fields: id(8) + owner(8) + layer(2) + key_dim(4) + value_dim(4) + pos_dim(4)
275    //   + token_span(16) + created_at(8) + updated_at(8) + importance(4) + tags(4) + tier(1)
276    //   + key_scales_len(4) + key_data_len(4) + val_scales_len(4) + val_data_len(4)
277    let fixed = 8 + 8 + 2 + 4 + 4 + 4 + 16 + 8 + 8 + 4 + 4 + 1 + 4 + 4 + 4 + 4;
278    let variable = key_q.scales.len() * 4
279        + key_q.data.len()
280        + val_q.scales.len() * 4
281        + val_q.data.len()
282        + pos_len * 4;
283    fixed + variable
284}

288/// Serialize a single `MemoryCell` to a writer (Q4 quantized).
289/// Returns the record body size (excluding the 4-byte `record_len` prefix).
290fn write_cell_record(w: &mut impl Write, cell: &MemoryCell) -> io::Result<u64> {
291    let key_q = Q4::quantize(&cell.key);
292    let val_q = Q4::quantize(&cell.value);
293
294    let record_bytes = compute_record_size(&key_q, &val_q, cell.pos_encoding.len());
295    let record_len: u32 = record_bytes.try_into().map_err(|_| {
296        io::Error::new(
297            io::ErrorKind::InvalidInput,
298            format!("record size {record_bytes} exceeds u32::MAX"),
299        )
300    })?;
301    w.write_all(&record_len.to_le_bytes())?;
302
303    // Fixed fields.
304    w.write_all(&cell.id.to_le_bytes())?;
305    w.write_all(&cell.owner.to_le_bytes())?;
306    w.write_all(&cell.layer.to_le_bytes())?;
307    w.write_all(&(cell.key.len() as u32).to_le_bytes())?;
308    w.write_all(&(cell.value.len() as u32).to_le_bytes())?;
309    w.write_all(&(cell.pos_encoding.len() as u32).to_le_bytes())?;
310    w.write_all(&cell.token_span.0.to_le_bytes())?;
311    w.write_all(&cell.token_span.1.to_le_bytes())?;
312    w.write_all(&cell.meta.created_at.to_le_bytes())?;
313    w.write_all(&cell.meta.updated_at.to_le_bytes())?;
314    w.write_all(&cell.meta.importance.to_le_bytes())?;
315    w.write_all(&cell.meta.tags.to_le_bytes())?;
316    w.write_all(&[cell.meta.tier as u8])?;
317
318    // Quantized key.
319    w.write_all(&(key_q.scales.len() as u32).to_le_bytes())?;
320    w.write_all(&(key_q.data.len() as u32).to_le_bytes())?;
321    // Quantized value.
322    w.write_all(&(val_q.scales.len() as u32).to_le_bytes())?;
323    w.write_all(&(val_q.data.len() as u32).to_le_bytes())?;
324
325    // Variable-length data.
326    write_f32_slice(w, &key_q.scales)?;
327    w.write_all(&key_q.data)?;
328    write_f32_slice(w, &val_q.scales)?;
329    w.write_all(&val_q.data)?;
330    write_f32_slice(w, &cell.pos_encoding)?;
331
332    Ok(record_bytes as u64)
333}

337fn write_f32_slice(w: &mut impl Write, values: &[f32]) -> io::Result<()> {
338    for &v in values {
339        w.write_all(&v.to_le_bytes())?;
340    }
341    Ok(())
342}

374fn read_f32_vec(r: &mut impl Read, count: usize) -> io::Result<Vec<f32>> {
375    let mut buf = vec![0u8; count * 4];
376    r.read_exact(&mut buf)?;
377    Ok(buf.chunks_exact(4).map(|c| f32::from_le_bytes([c[0], c[1], c[2], c[3]])).collect())
378}

380fn read_bytes(r: &mut impl Read, count: usize) -> io::Result<Vec<u8>> {
381    let mut buf = vec![0u8; count];
382    r.read_exact(&mut buf)?;
383    Ok(buf)
