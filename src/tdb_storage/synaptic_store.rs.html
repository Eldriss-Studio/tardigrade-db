1//! Append-only file store for `SynapticBankEntry` (`LoRA` adapters).
2//!
3//! Repository pattern: mirrors `BlockPool` conventions — binary records,
4//! length-prefixed, `sync_data()` for durability, scan-on-open for recovery.
5//!
6//! ## Record format (binary, little-endian)
7//!
8//! ```text
9//! [record_len: u32]    — total bytes excluding this field
10//! [id: u64]            — synaptic entry ID
11//! [owner: u64]         — agent/user owner
12//! [rank: u32]          — LoRA rank
13//! [d_model: u32]       — model dimension
14//! [scale: u16]         — f16 scaling factor
15//! [last_used: u64]     — timestamp
16//! [quality: f32]       — quality score
17//! [lora_a: ...]        — rank * d_model * 2 bytes (f16)
18//! [lora_b: ...]        — d_model * rank * 2 bytes (f16)
19//! ```
20
21use std::fs::{File, OpenOptions};
22use std::io::{self, BufWriter, Read, Seek, SeekFrom, Write};
23use std::path::{Path, PathBuf};
24
25use half::f16;
26use tdb_core::synaptic_bank::SynapticBankEntry;
27use tdb_core::types::OwnerId;

29const SYNAPTIC_FILE_NAME: &str = "synaptic.tdb";
30
31/// Repository for persisting and loading `SynapticBankEntry` records.
32#[derive(Debug)]
33pub struct SynapticStore {
34    path: PathBuf,
35    /// In-memory index: (id, owner, `byte_offset`).
36    index: Vec<(u64, OwnerId, u64)>,
37}
38
39impl SynapticStore {
40    /// Open or create a synaptic store at the given directory.
41    pub fn open(dir: &Path) -> io::Result<Self> {
42        std::fs::create_dir_all(dir)?;
43        let path = dir.join(SYNAPTIC_FILE_NAME);
44
45        // Touch file if it doesn't exist.
46        OpenOptions::new().create(true).append(true).open(&path)?;
47
48        // Scan to build index.
49        let index = scan_entries(&path)?;
50
51        Ok(Self { path, index })
52    }
53
54    /// Append a `SynapticBankEntry` to the store.
55    pub fn append(&mut self, entry: &SynapticBankEntry) -> io::Result<()> {
56        let file = OpenOptions::new().append(true).open(&self.path)?;
57        let offset = file.metadata()?.len();
58        let mut w = BufWriter::new(file);
59
60        let lora_bytes = entry.lora_a.len() * 2 + entry.lora_b.len() * 2;
61        // Fixed: id(8) + owner(8) + rank(4) + d_model(4) + scale(2) + last_used(8) + quality(4) = 38
62        let record_len: u32 = (38 + lora_bytes)
63            .try_into()
64            .map_err(|_| io::Error::new(io::ErrorKind::InvalidInput, "record too large"))?;
65
66        w.write_all(&record_len.to_le_bytes())?;
67        w.write_all(&entry.id.to_le_bytes())?;
68        w.write_all(&entry.owner.to_le_bytes())?;
69        w.write_all(&entry.rank.to_le_bytes())?;
70        w.write_all(&entry.d_model.to_le_bytes())?;
71        w.write_all(&entry.scale.to_le_bytes())?;
72        w.write_all(&entry.last_used.to_le_bytes())?;
73        w.write_all(&entry.quality.to_le_bytes())?;
74
75        write_f16_slice(&mut w, &entry.lora_a)?;
76        write_f16_slice(&mut w, &entry.lora_b)?;
77
78        let inner = w.into_inner().map_err(std::io::IntoInnerError::into_error)?;
79        inner.sync_data()?;
80
81        self.index.push((entry.id, entry.owner, offset));
82        Ok(())
83    }
84
85    /// Load all entries belonging to a specific owner.
86    pub fn load_by_owner(&self, owner: OwnerId) -> io::Result<Vec<SynapticBankEntry>> {
87        let offsets: Vec<u64> = self
88            .index
89            .iter()
90            .filter(|(_, o, _)| *o == owner)
91            .map(|(_, _, offset)| *offset)
92            .collect();
93
94        let mut results = Vec::with_capacity(offsets.len());
95        for offset in offsets {
96            results.push(read_entry_at(&self.path, offset)?);
97        }
98        Ok(results)
99    }
100
101    /// Number of entries in the store.
102    pub fn entry_count(&self) -> usize {
103        self.index.len()
104    }
105}

107fn scan_entries(path: &Path) -> io::Result<Vec<(u64, OwnerId, u64)>> {
108    let mut file = File::open(path)?;
109    let file_len = file.metadata()?.len();
110    let mut entries = Vec::new();
111    let mut pos = 0u64;
112
113    while pos + 4 <= file_len {
114        let record_len = match read_u32(&mut file) {
115            Ok(len) => len as u64,
116            Err(_) => break,
117        };
118
119        if pos + 4 + record_len > file_len {
120            break;
121        }
122
123        let Ok(id) = read_u64(&mut file) else {
124            break;
125        };
126        let Ok(owner) = read_u64(&mut file) else {
127            break;
128        };
129
130        entries.push((id, owner, pos));
131
132        // Skip rest of record: already read 4 (record_len) + 8 (id) + 8 (owner) = 20.
133        let remaining = record_len.saturating_sub(16);
134        if file.seek(SeekFrom::Current(i64::try_from(remaining).unwrap_or(i64::MAX))).is_err() {
135            break;
136        }
137        pos += 4 + record_len;
138    }
139
140    Ok(entries)
141}

143fn read_entry_at(path: &Path, offset: u64) -> io::Result<SynapticBankEntry> {
144    let mut file = File::open(path)?;
145    file.seek(SeekFrom::Start(offset))?;
146
147    let _record_len = read_u32(&mut file)?;
148    let id = read_u64(&mut file)?;
149    let owner = read_u64(&mut file)?;
150    let rank = read_u32(&mut file)?;
151    let d_model = read_u32(&mut file)?;
152
153    let mut scale_buf = [0u8; 2];
154    file.read_exact(&mut scale_buf)?;
155    let scale = f16::from_le_bytes(scale_buf);
156
157    let last_used = read_u64(&mut file)?;
158    let quality = read_f32(&mut file)?;
159
160    let lora_len = (rank as usize) * (d_model as usize);
161    let lora_a = read_f16_vec(&mut file, lora_len)?;
162    let lora_b = read_f16_vec(&mut file, lora_len)?;
163
164    Ok(SynapticBankEntry { id, owner, lora_a, lora_b, scale, rank, d_model, last_used, quality })
165}
166
167fn write_f16_slice(w: &mut impl Write, values: &[f16]) -> io::Result<()> {
168    for &v in values {
169        w.write_all(&v.to_le_bytes())?;
170    }
171    Ok(())
172}
173
174fn read_f16_vec(r: &mut impl Read, count: usize) -> io::Result<Vec<f16>> {
175    let mut buf = vec![0u8; count * 2];
176    r.read_exact(&mut buf)?;
177    Ok(buf.chunks_exact(2).map(|c| f16::from_le_bytes([c[0], c[1]])).collect())
178}
179
180fn read_u32(r: &mut impl Read) -> io::Result<u32> {
181    let mut buf = [0u8; 4];
182    r.read_exact(&mut buf)?;
183    Ok(u32::from_le_bytes(buf))
184}
185
186fn read_u64(r: &mut impl Read) -> io::Result<u64> {
187    let mut buf = [0u8; 8];
188    r.read_exact(&mut buf)?;
189    Ok(u64::from_le_bytes(buf))
190}
191
192fn read_f32(r: &mut impl Read) -> io::Result<f32> {
193    let mut buf = [0u8; 4];
194    r.read_exact(&mut buf)?;
195    Ok(f32::from_le_bytes(buf))
