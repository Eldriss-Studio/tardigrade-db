1//! Block pool — the Repository abstraction over segmented storage.
2//!
3//! Provides `append` and `get` operations over a collection of segment files,
4//! with an in-memory index mapping `CellId` → (`segment_id`, `byte_offset`).
5//! The index is rebuilt from segment files on open (recovery).
6
7use std::collections::{BTreeMap, HashSet};
8use std::path::{Path, PathBuf};
9
10use tdb_core::CellId;
11use tdb_core::error::{Result, TardigradeError};
12use tdb_core::memory_cell::MemoryCell;
13
14use crate::segment::{RecordLocation, Segment, list_segments, scan_segment, segment_path};

16/// Default segment size threshold: 256 MB.
17const DEFAULT_SEGMENT_SIZE: u64 = 256 * 1024 * 1024;
18
19/// Segments with a live-cell ratio below this threshold are candidates for compaction.
20const COMPACTION_LIVE_RATIO_THRESHOLD: f64 = 0.5;
21
22/// Result of a compaction operation.
23#[derive(Debug, Clone, Default)]
24pub struct CompactionResult {
25    pub segments_compacted: usize,
26    pub cells_moved: usize,
27    pub bytes_reclaimed: u64,
28}
29
30struct CompactJob {
31    seg_id: u32,
32    cells: Vec<MemoryCell>,
33    file_size: u64,
34}

36/// Repository over segmented, append-only storage for memory cells.
37///
38/// The index is held in memory (`BTreeMap`) and rebuilt from segment files on open.
39/// Segments are append-only; when the active segment exceeds the size threshold,
40/// a new segment is created.
41#[derive(Debug)]
42pub struct BlockPool {
43    dir: PathBuf,
44    segments: Vec<Segment>,
45    index: BTreeMap<CellId, RecordLocation>,
46    segment_size_threshold: u64,
47}
48
49impl BlockPool {
50    /// Open or create a block pool at the given directory path.
51    /// Rebuilds the in-memory index by scanning existing segments.
52    pub fn open(dir: &Path) -> Result<Self> {
53        Self::open_with_segment_size(dir, DEFAULT_SEGMENT_SIZE)
54    }
55
56    /// Open with a custom segment size threshold (useful for testing).
57    pub fn open_with_segment_size(dir: &Path, segment_size_threshold: u64) -> Result<Self> {
58        std::fs::create_dir_all(dir)?;
59
60        let segment_ids = list_segments(dir)?;
61        let mut segments = Vec::new();
62        let mut index = BTreeMap::new();
63
64        for &seg_id in &segment_ids {
65            let segment = Segment::open(dir, seg_id)?;
66            let entries = scan_segment(dir, seg_id)?;
67            for (cell_id, byte_offset) in entries {
68                index.insert(cell_id, RecordLocation { segment_id: seg_id, byte_offset });
69            }
70            segments.push(segment);
71        }
72
73        // If no segments exist, create the first one.
74        if segments.is_empty() {
75            segments.push(Segment::create(dir, 0)?);
76        }
77
78        Ok(Self { dir: dir.to_path_buf(), segments, index, segment_size_threshold })
79    }

81    /// Re-scan segment files on disk and merge any new entries into the
82    /// in-memory index. Picks up cells written by another `BlockPool`
83    /// handle at the same path.
84    ///
85    /// Idempotent: re-scanning a segment that hasn't grown leaves the
86    /// index unchanged. New segment files (if another writer rolled over)
87    /// are opened and added to `self.segments`. Existing segment file
88    /// handles are not re-opened.
89    pub fn refresh_index(&mut self) -> Result<()> {
90        let segment_ids = list_segments(&self.dir)?;
91
92        // Pick up any newly-created segments (other writer rolled over).
93        let known: std::collections::HashSet<u32> = self.segments.iter().map(Segment::id).collect();
94        for &seg_id in &segment_ids {
95            if !known.contains(&seg_id) {
96                self.segments.push(Segment::open(&self.dir, seg_id)?);
97            }
98        }
99
100        // Re-scan every segment (cheap — header reads only) and merge into
101        // the index. `scan_segment` is idempotent on append-only files.
102        for &seg_id in &segment_ids {
103            let entries = scan_segment(&self.dir, seg_id)?;
104            for (cell_id, byte_offset) in entries {
105                self.index
106                    .entry(cell_id)
107                    .or_insert(RecordLocation { segment_id: seg_id, byte_offset });
108            }
109        }
110
111        Ok(())
112    }

114    /// Append a memory cell to the pool. Returns the cell ID on success.
115    pub fn append(&mut self, cell: &MemoryCell) -> Result<CellId> {
116        self.ensure_active_segment_has_capacity()?;
117
118        let active = self.active_segment_mut()?;
119        let seg_id = active.id();
120        let byte_offset = active.append(cell)?;
121
122        self.index.insert(cell.id, RecordLocation { segment_id: seg_id, byte_offset });
123
124        Ok(cell.id)
125    }
126
127    /// Append multiple cells in a single write + single fsync (Write-Behind Buffer).
128    ///
129    /// All cells are written to the active segment and durably committed with
130    /// one `sync_data()` call. Returns the cell IDs.
131    pub fn append_batch(&mut self, cells: &[MemoryCell]) -> Result<Vec<CellId>> {
132        if cells.is_empty() {
133            return Ok(Vec::new());
134        }
135
136        self.ensure_active_segment_has_capacity()?;
137
138        let active = self.active_segment_mut()?;
139        let seg_id = active.id();
140        let offsets = active.append_batch(cells)?;
141
142        let mut ids = Vec::with_capacity(cells.len());
143        for (cell, byte_offset) in cells.iter().zip(offsets) {
144            self.index.insert(cell.id, RecordLocation { segment_id: seg_id, byte_offset });
145            ids.push(cell.id);
146        }
147
148        Ok(ids)
149    }

151    /// Retrieve a memory cell by its ID.
152    pub fn get(&self, cell_id: CellId) -> Result<MemoryCell> {
153        let loc = self.index.get(&cell_id).ok_or(TardigradeError::CellNotFound(cell_id))?;
154
155        let segment = self
156            .segments
157            .iter()
158            .find(|s| s.id() == loc.segment_id)
159            .ok_or(TardigradeError::CellNotFound(cell_id))?;
160
161        Ok(segment.read_at(loc.byte_offset)?)
162    }

164    /// Number of segment files in this pool.
165    pub fn segment_count(&self) -> usize {
166        self.segments.len()
167    }
168
169    /// Number of cells tracked in the index.
170    pub fn cell_count(&self) -> usize {
171        self.index.len()
172    }

174    /// Iterate over all persisted cell IDs (sorted, from the in-memory index).
175    /// Used by `Engine::open()` to rebuild derived state from disk (Memento pattern).
176    pub fn iter_cell_ids(&self) -> impl Iterator<Item = CellId> + '_ {
177        self.index.keys().copied()
178    }

180    /// If the active segment exceeds the threshold, create a new one.
181    fn ensure_active_segment_has_capacity(&mut self) -> Result<()> {
182        let needs_rollover =
183            self.segments.last().is_some_and(|s| s.size() >= self.segment_size_threshold);
184
185        if needs_rollover {
186            let last = self
187                .segments
188                .last()
189                .ok_or(TardigradeError::SegmentFull { path: self.dir.display().to_string() })?;
190            let new_id = last.id() + 1;
191            let new_segment = Segment::create(&self.dir, new_id)?;
192            self.segments.push(new_segment);
193        }
194        Ok(())
195    }

197    /// Compact segments by rewriting live cells and deleting dead ones (Mark-Sweep).
198    ///
199    /// Non-active segments where the ratio of live cells falls below
200    /// `COMPACTION_LIVE_RATIO_THRESHOLD` are rewritten: live cells are
201    /// appended to the active segment, then the old segment file is deleted.
202    ///
203    /// Crash-safe: new cells are fsynced before old segment deletion. If a
204    /// crash occurs between write and delete, the next `open()` rebuilds
205    /// from all segments — duplicates are harmless (index deduplicates).
206    pub fn compact(&mut self, live_cell_ids: &HashSet<CellId>) -> Result<CompactionResult> {
207        let mut result = CompactionResult::default();
208
209        if self.segments.len() <= 1 {
210            return Ok(result);
211        }
212
213        let active_seg_id = self.segments.last().map_or(0, Segment::id);
214
215        let mut jobs: Vec<CompactJob> = Vec::new();
216
217        for seg_idx in 0..self.segments.len() {
218            let seg_id = self.segments[seg_idx].id();
219            if seg_id == active_seg_id {
220                continue;
221            }
222
223            let entries =
224                scan_segment(&self.dir, seg_id).map_err(|e| TardigradeError::Io { source: e })?;
225            if entries.is_empty() {
226                continue;
227            }
228
229            let live_count = entries.iter().filter(|(cid, _)| live_cell_ids.contains(cid)).count();
230            let total = entries.len();
231            let live_ratio = live_count as f64 / total as f64;
232
233            if live_ratio >= COMPACTION_LIVE_RATIO_THRESHOLD {
234                continue;
235            }
236
237            let mut cells = Vec::with_capacity(live_count);
238            for (cell_id, byte_offset) in &entries {
239                if live_cell_ids.contains(cell_id) {
240                    let cell = self.segments[seg_idx]
241                        .read_at(*byte_offset)
242                        .map_err(|e| TardigradeError::Io { source: e })?;
243                    cells.push(cell);
244                }
245            }
246
247            let seg_path = segment_path(&self.dir, seg_id);
248            let file_size = std::fs::metadata(&seg_path).map_or(0, |m| m.len());
249
250            jobs.push(CompactJob { seg_id, cells, file_size });
251        }
252
253        if jobs.is_empty() {
254            return Ok(result);
255        }
256
257        for job in &jobs {
258            if !job.cells.is_empty() {
259                self.ensure_active_segment_has_capacity()?;
260                let active = self.active_segment_mut()?;
261                let new_seg_id = active.id();
262                let offsets = active
263                    .append_batch(&job.cells)
264                    .map_err(|e| TardigradeError::Io { source: e })?;
265
266                for (cell, offset) in job.cells.iter().zip(offsets) {
267                    self.index.insert(
268                        cell.id,
269                        RecordLocation { segment_id: new_seg_id, byte_offset: offset },
270                    );
271                }
272                result.cells_moved += job.cells.len();
273            }
274
275            result.bytes_reclaimed += job.file_size;
276        }
277
278        let compacted_ids: HashSet<u32> = jobs.iter().map(|j| j.seg_id).collect();
279        self.index.retain(|_, loc| !compacted_ids.contains(&loc.segment_id));
280
281        for seg_id in &compacted_ids {
282            let path = segment_path(&self.dir, *seg_id);
283            std::fs::remove_file(&path).map_err(|e| TardigradeError::Io { source: e })?;
284        }
285
286        self.segments.retain(|s| !compacted_ids.contains(&s.id()));
287        result.segments_compacted = compacted_ids.len();
288
289        Ok(result)
290    }
291
292    fn active_segment_mut(&mut self) -> Result<&mut Segment> {
293        self.segments
294            .last_mut()
295            .ok_or(TardigradeError::SegmentFull { path: self.dir.display().to_string() })
296    }
