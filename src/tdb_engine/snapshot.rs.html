1//! Portable snapshot/restore for the engine.
2//!
3//! Bundles the engine's persistent state into a single tar archive
4//! with a forensic-friendly manifest header. The format is adapted
5//! from `SpacetimeDB`'s `crates/snapshot` (magic bytes + version +
6//! offset-indexed lookup) with `TardigradeDB`-specific additions
7//! (codec versioning so snapshots survive future compressor swaps).
8//!
9//! ## On-disk layout
10//!
11//! ```text
12//! <out_path>.tar
13//!   manifest.json           — versioned manifest (see SnapshotManifest)
14//!   engine_state/           — verbatim copy of the engine's working dir
15//!     <block-pool segments, WAL, text store, deletion log, ...>
16//! ```
17//!
18//! ## Reliability contract
19//!
20//! - **Durability.** [`crate::engine::Engine::snapshot`] calls [`crate::engine::Engine::flush`] before
21//!   tarring; the archive therefore reflects state at the
22//!   `durable_offset` of flush completion.
23//! - **Recovery.** [`crate::engine::Engine::restore_from`] is equivalent to a fresh
24//!   [`crate::engine::Engine::open`] of the same directory after extraction. All
25//!   existing replay rules apply unchanged.
26//! - **Versioning.** Restore refuses unknown `format_version` and
27//!   incompatible `codecs` with structured errors. Forward-compat is
28//!   an explicit conversion tool, not implicit.
29//! - **Integrity.** A SHA-256 of the `engine_state/` payload is
30//!   stored in the manifest; restore verifies it before extracting.
31

43/// Magic marker placed at the head of every `TardigradeDB` snapshot
44/// manifest. Lets external tooling (and human forensics) distinguish
45/// our archives from arbitrary tarballs.
46pub const SNAPSHOT_MAGIC: &str = "tdb!";
47
48/// Current snapshot format version. Increment on any breaking change
49/// to the manifest schema, archive layout, or required state files.
50/// Older readers refuse newer versions; newer readers may accept
51/// older versions when the migration is trivial.
52pub const SNAPSHOT_FORMAT_VERSION: u8 = 1;
53
54/// Versioned identifier for the quantization codec currently in use.
55/// Snapshots embed this so a future engine that drops Q4 support can
56/// refuse Q4 snapshots cleanly instead of producing garbage reads.
57pub const SNAPSHOT_QUANT_CODEC: &str = "q4";
58
59/// Versioned identifier for the per-token retrieval-key codec.
60pub const SNAPSHOT_KEY_CODEC: &str = "top5avg";
61
62/// Top-level directory inside the tar archive holding the engine's
63/// state files. Kept stable across versions so restore tooling can
64/// inspect contents without parsing the manifest.
65pub const SNAPSHOT_PAYLOAD_DIR: &str = "engine_state";
66
67/// Filename of the manifest inside the tar archive. Always at the
68/// archive root, always read first.
69pub const SNAPSHOT_MANIFEST_NAME: &str = "manifest.json";
70
71/// Upper bound (bytes) on the manifest size during restore. The
72/// real manifest is ~500 bytes; 1 MiB is a generous ceiling that
73/// guards against a corrupt or hostile archive declaring an
74/// enormous entry and exhausting RAM via [`Read::read_to_end`].
75const MAX_MANIFEST_BYTES: u64 = 1024 * 1024;
76
77/// Upper bound (bytes) on a single payload file during restore.
78/// 16 GiB accommodates the largest plausible `BlockPool` segment
79/// (`segment_size` is 256 MiB by default; we leave 64× headroom for
80/// future configurations) while still bounding worst-case RAM use.
81const MAX_PAYLOAD_FILE_BYTES: u64 = 16 * 1024 * 1024 * 1024;

85/// Versioned metadata header written at the root of every snapshot
86/// archive. Read first by [`crate::engine::Engine::restore_from`] to validate
87/// compatibility before extracting the payload.
88#[derive(Debug, Clone, Serialize, Deserialize)]
89pub struct SnapshotManifest {
90    /// Magic marker; must equal [`SNAPSHOT_MAGIC`].
91    pub magic: String,
92    /// Format version; restore refuses unknown values.
93    pub format_version: u8,
94    /// RFC 3339 timestamp of snapshot creation.
95    pub created_at: String,
96    /// Codec identifiers for the storage formats embedded in the
97    /// payload. Restore refuses snapshots whose codecs are not
98    /// supported by the current engine.
99    pub codecs: SnapshotCodecs,
100    /// Cheap stats useful for forensics + repository lookup.
101    pub stats: SnapshotStats,
102    /// SHA-256 (hex-encoded) of the tarred ``engine_state/`` payload.
103    /// Computed *over the engine state files only* — not over the
104    /// manifest itself. Verified by [`crate::engine::Engine::restore_from`].
105    pub sha256: String,
106}

108/// Codec identifiers embedded in a snapshot manifest. Used to detect
109/// format swaps between snapshot-write and snapshot-restore.
110#[derive(Debug, Clone, Serialize, Deserialize)]
111pub struct SnapshotCodecs {
112    /// Quantization codec (e.g. ``"q4"``).
113    pub quantization: String,
114    /// Per-token retrieval-key codec (e.g. ``"top5avg"``).
115    pub key: String,
116}

118/// Snapshot stats — handy for the repository's `latest()` lookup and
119/// for displaying in admin UIs without unpacking the archive.
120#[derive(Debug, Clone, Default, Serialize, Deserialize)]
121pub struct SnapshotStats {
122    /// Total packs in the engine at snapshot time.
123    pub pack_count: usize,
124    /// Distinct owners with at least one pack.
125    pub owner_count: usize,
126}

128impl SnapshotManifest {
129    /// Construct a manifest with the current format constants.
130    /// `sha256` and `stats` must be filled by the caller.
131    #[must_use]
132    pub fn current(sha256: String, stats: SnapshotStats) -> Self {
133        Self {
134            magic: SNAPSHOT_MAGIC.to_string(),
135            format_version: SNAPSHOT_FORMAT_VERSION,
136            created_at: rfc3339_now(),
137            codecs: SnapshotCodecs {
138                quantization: SNAPSHOT_QUANT_CODEC.to_string(),
139                key: SNAPSHOT_KEY_CODEC.to_string(),
140            },
141            stats,
142            sha256,
143        }
144    }
145}

147/// Format an RFC 3339 timestamp without pulling in `chrono`. Returns
148/// seconds since the Unix epoch as a string; sufficient for forensic
149/// ordering and human readability inside admin tooling.
150fn rfc3339_now() -> String {
151    let secs = std::time::SystemTime::now()
152        .duration_since(std::time::UNIX_EPOCH)
153        .map_or(0, |d| d.as_secs());
154    // Compact ISO-ish: ``1970-01-01T00:00:00Z + N seconds``. We do
155    // not need calendar conversion at this layer; downstream tooling
156    // can pretty-print if needed.
157    format!("@{secs}")
158}

162/// Bundle the engine's working directory into a tar archive at
163/// ``out_path``. Returns the manifest by value so callers can log or
164/// persist it without re-reading the archive.
165///
166/// The caller must have already flushed the engine. [`crate::engine::Engine::snapshot`]
167/// in `engine.rs` is the proper public entry point and handles flush.
168///
169/// # Errors
170///
171/// Returns [`TardigradeError::Io`] on disk failures and a
172/// [`TardigradeError::SnapshotIntegrity`] if the manifest cannot be
173/// serialized (which would indicate a bug, not user input).
174pub fn write_snapshot(
175    engine_dir: &Path,
176    out_path: &Path,
177    stats: SnapshotStats,
178) -> Result<SnapshotManifest> {
179    // Refuse the foot-gun where ``out_path`` sits inside
180    // ``engine_dir`` — the tar walker would read its own growing
181    // output and exhaust disk. Caller must write the snapshot
182    // somewhere outside the engine state.
183    if let (Ok(out_abs), Ok(engine_abs)) = (
184        out_path.canonicalize().or_else(|_| {
185            // The output usually doesn't exist yet; resolve its parent.
186            out_path.parent().map_or_else(
187                || Ok(out_path.to_path_buf()),
188                |p| p.canonicalize().map(|abs| abs.join(out_path.file_name().unwrap_or_default())),
189            )
190        }),
191        engine_dir.canonicalize(),
192    ) && out_abs.starts_with(&engine_abs)
193    {
194        return Err(TardigradeError::SnapshotIntegrity(format!(
195            "out_path ({}) is inside engine_dir ({}); pick a path outside the engine directory",
196            out_path.display(),
197            engine_dir.display(),
198        )));
199    }
200
201    let mut hasher = Sha256::new();
202
203    // Phase 1 — hash the payload first so we can include the digest
204    // in the manifest. We walk the same set of files we'll
205    // subsequently tar so the digest matches what restore will see.
206    walk_files_for_hash(engine_dir, &mut hasher)?;
207    let sha256 = hex_lower(&hasher.finalize());
208
209    let manifest = SnapshotManifest::current(sha256, stats);
210    let manifest_json = serde_json::to_vec_pretty(&manifest)
211        .map_err(|e| TardigradeError::SnapshotIntegrity(format!("manifest serialize: {e}")))?;
212
213    // Phase 2 — write the tar archive: manifest.json first, then the
214    // engine_state/ payload.
215    let file = File::create(out_path).map_err(|e| TardigradeError::Io { source: e })?;
216    let mut tar = Builder::new(file);
217    tar.mode(tar::HeaderMode::Deterministic);
218
219    // Manifest entry.
220    let mut header = tar::Header::new_gnu();
221    header.set_path(SNAPSHOT_MANIFEST_NAME).map_err(|e| TardigradeError::Io { source: e })?;
222    header.set_size(manifest_json.len() as u64);
223    header.set_mode(0o644);
224    header.set_cksum();
225    tar.append(&header, &manifest_json[..]).map_err(|e| TardigradeError::Io { source: e })?;
226
227    // Payload: copy every regular file under engine_dir into
228    // engine_state/ inside the archive.
229    tar.append_dir_all(SNAPSHOT_PAYLOAD_DIR, engine_dir)
230        .map_err(|e| TardigradeError::Io { source: e })?;
231
232    tar.finish().map_err(|e| TardigradeError::Io { source: e })?;
233    Ok(manifest)
234}

236/// Read just the manifest from a snapshot archive without
237/// extracting the payload or verifying the integrity hash.
238///
239/// Useful for listing operations that only need metadata (label
240/// directories, repository enumeration). Magic and format version
241/// are still validated — a tar that lies about being a
242/// `TardigradeDB` snapshot is refused here too.
243///
244/// # Errors
245///
246/// - [`TardigradeError::NotATardigradeSnapshot`] — magic missing.
247/// - [`TardigradeError::UnsupportedFormatVersion`] — unknown version.
248/// - [`TardigradeError::Io`] — disk failure.
249pub fn read_manifest_only(in_path: &Path) -> Result<SnapshotManifest> {
250    let file = File::open(in_path).map_err(|e| TardigradeError::Io { source: e })?;
251    let mut archive = Archive::new(file);
252    for entry in archive.entries().map_err(|e| TardigradeError::Io { source: e })? {
253        let mut entry = entry.map_err(|e| TardigradeError::Io { source: e })?;
254        let path = entry.path().map_err(|e| TardigradeError::Io { source: e })?.to_path_buf();
255        if path.as_os_str() != SNAPSHOT_MANIFEST_NAME {
256            continue;
257        }
258        let buf = read_entry_bounded(&mut entry, MAX_MANIFEST_BYTES)?;
259        let manifest: SnapshotManifest = serde_json::from_slice(&buf).map_err(|e| {
260            TardigradeError::NotATardigradeSnapshot { reason: format!("manifest parse: {e}") }
261        })?;
262        if manifest.magic != SNAPSHOT_MAGIC {
263            return Err(TardigradeError::NotATardigradeSnapshot {
264                reason: format!("unexpected magic: {:?}", manifest.magic),
265            });
266        }
267        if manifest.format_version != SNAPSHOT_FORMAT_VERSION {
268            return Err(TardigradeError::UnsupportedFormatVersion {
269                found: manifest.format_version,
270                supported: SNAPSHOT_FORMAT_VERSION,
271            });
272        }
273        return Ok(manifest);
274    }
275    Err(TardigradeError::NotATardigradeSnapshot { reason: "manifest.json missing".to_string() })
276}

278/// Restore a snapshot archive into ``target_dir`` (which must be
279/// empty or non-existent) and validate its manifest. The caller
280/// invokes [`crate::engine::Engine::open`] on ``target_dir`` afterwards.
281///
282/// Returns the parsed manifest so callers can inspect or log the
283/// snapshot's metadata.
284///
285/// # Errors
286///
287/// - [`TardigradeError::NotATardigradeSnapshot`] — magic missing.
288/// - [`TardigradeError::UnsupportedFormatVersion`] — unknown version.
289/// - [`TardigradeError::SnapshotCodecMismatch`] — codec incompatible.
290/// - [`TardigradeError::SnapshotIntegrity`] — SHA-256 mismatch.
291/// - [`TardigradeError::Io`] — disk failure.
292pub fn read_snapshot(in_path: &Path, target_dir: &Path) -> Result<SnapshotManifest> {
293    let file = File::open(in_path).map_err(|e| TardigradeError::Io { source: e })?;
294    let mut archive = Archive::new(file);
295
296    // First entry: manifest. Pull it out independently so we can
297    // refuse the snapshot before extracting any payload.
298    let mut manifest_bytes: Option<Vec<u8>> = None;
299    let mut payload_entries: Vec<(PathBuf, Vec<u8>)> = Vec::new();
300
301    for entry in archive.entries().map_err(|e| TardigradeError::Io { source: e })? {
302        let mut entry = entry.map_err(|e| TardigradeError::Io { source: e })?;
303        let path = entry.path().map_err(|e| TardigradeError::Io { source: e })?.to_path_buf();
304
305        if path.as_os_str() == SNAPSHOT_MANIFEST_NAME {
306            let buf = read_entry_bounded(&mut entry, MAX_MANIFEST_BYTES)?;
307            manifest_bytes = Some(buf);
308            continue;
309        }
310        // Only regular files inside engine_state/ are part of the
311        // payload; tar Directory entries are skipped (we re-create
312        // dirs on extract).
313        if !entry.header().entry_type().is_file() {
314            continue;
315        }
316        if !path.starts_with(SNAPSHOT_PAYLOAD_DIR) {
317            continue;
318        }
319        let buf = read_entry_bounded(&mut entry, MAX_PAYLOAD_FILE_BYTES)?;
320        payload_entries.push((path, buf));
321    }
322
323    let manifest_bytes = manifest_bytes.ok_or_else(|| TardigradeError::NotATardigradeSnapshot {
324        reason: "manifest.json missing".to_string(),
325    })?;
326    let manifest: SnapshotManifest = serde_json::from_slice(&manifest_bytes).map_err(|e| {
327        TardigradeError::NotATardigradeSnapshot { reason: format!("manifest parse: {e}") }
328    })?;
329
330    if manifest.magic != SNAPSHOT_MAGIC {
331        return Err(TardigradeError::NotATardigradeSnapshot {
332            reason: format!("unexpected magic: {:?}", manifest.magic),
333        });
334    }
335    if manifest.format_version != SNAPSHOT_FORMAT_VERSION {
336        return Err(TardigradeError::UnsupportedFormatVersion {
337            found: manifest.format_version,
338            supported: SNAPSHOT_FORMAT_VERSION,
339        });
340    }
341    if manifest.codecs.quantization != SNAPSHOT_QUANT_CODEC {
342        return Err(TardigradeError::SnapshotCodecMismatch {
343            field: "quantization".to_string(),
344            snapshot: manifest.codecs.quantization.clone(),
345            engine: SNAPSHOT_QUANT_CODEC.to_string(),
346        });
347    }
348    if manifest.codecs.key != SNAPSHOT_KEY_CODEC {
349        return Err(TardigradeError::SnapshotCodecMismatch {
350            field: "key".to_string(),
351            snapshot: manifest.codecs.key.clone(),
352            engine: SNAPSHOT_KEY_CODEC.to_string(),
353        });
354    }
355
356    // Verify the payload SHA-256 against the manifest. Both write
357    // and read sides hash in *lexicographic* order of the relative
358    // path so the digest is order-independent of the underlying
359    // filesystem (OS readdir order leaked through tar entries
360    // otherwise and produced spurious mismatches).
361    payload_entries.sort_by(|a, b| a.0.cmp(&b.0));
362    let mut hasher = Sha256::new();
363    for (path, bytes) in &payload_entries {
364        let rel = strip_payload_prefix(path);
365        hash_one(&mut hasher, rel.as_os_str(), bytes);
366    }
367    let computed = hex_lower(&hasher.finalize());
368    if computed != manifest.sha256 {
369        return Err(TardigradeError::SnapshotIntegrity(format!(
370            "sha256 mismatch: manifest={}, computed={}",
371            manifest.sha256, computed
372        )));
373    }
374
375    // Extract payload into target_dir.
376    fs::create_dir_all(target_dir).map_err(|e| TardigradeError::Io { source: e })?;
377    for (path, bytes) in payload_entries {
378        let rel = strip_payload_prefix(&path);
379        let dest = target_dir.join(rel);
380        if let Some(parent) = dest.parent() {
381            fs::create_dir_all(parent).map_err(|e| TardigradeError::Io { source: e })?;
382        }
383        let mut f = File::create(&dest).map_err(|e| TardigradeError::Io { source: e })?;
384        f.write_all(&bytes).map_err(|e| TardigradeError::Io { source: e })?;
385    }
386
387    Ok(manifest)
388}

392/// Walk every regular file under ``engine_dir`` in a stable order
393/// (lexicographic by relative path) and feed each path + contents
394/// into ``hasher``. The order must match the equivalent walk on the
395/// read side, otherwise the digest will mismatch.
396fn walk_files_for_hash(engine_dir: &Path, hasher: &mut Sha256) -> Result<()> {
397    let mut entries: Vec<PathBuf> = Vec::new();
398    collect_files(engine_dir, engine_dir, &mut entries)?;
399    entries.sort();
400    for rel in entries {
401        let abs = engine_dir.join(&rel);
402        let bytes = fs::read(&abs).map_err(|e| TardigradeError::Io { source: e })?;
403        hash_one(hasher, rel.as_os_str(), &bytes);
404    }
405    Ok(())
406}

408fn collect_files(root: &Path, dir: &Path, out: &mut Vec<PathBuf>) -> Result<()> {
409    for entry in fs::read_dir(dir).map_err(|e| TardigradeError::Io { source: e })? {
410        let entry = entry.map_err(|e| TardigradeError::Io { source: e })?;
411        let path = entry.path();
412        if path.is_dir() {
413            collect_files(root, &path, out)?;
414        } else if path.is_file() {
415            let rel = path.strip_prefix(root).unwrap_or(&path).to_path_buf();
416            out.push(rel);
417        }
418    }
419    Ok(())
420}

422fn strip_payload_prefix(path: &Path) -> PathBuf {
423    path.strip_prefix(SNAPSHOT_PAYLOAD_DIR).unwrap_or(path).to_path_buf()
424}

426/// Read up to `max_bytes` from `r` into a fresh `Vec<u8>`, erroring
427/// if more bytes are available. Provides the [`Read::read_to_end`]
428/// convenience without the OOM exposure on corrupt archives that
429/// declare absurdly large entry sizes.
430fn read_entry_bounded<R: Read>(r: &mut R, max_bytes: u64) -> Result<Vec<u8>> {
431    // Drain the entry into a Vec via a fixed scratch buffer; refuses
432    // (with a typed integrity error) any entry that exceeds
433    // `max_bytes`. Hand-rolled rather than `Read::read_to_end` so we
434    // never allocate beyond the cap on a corrupt or hostile archive.
435    const CHUNK: usize = 64 * 1024;
436    let mut buf: Vec<u8> = Vec::new();
437    let mut scratch = vec![0u8; CHUNK];
438    let max = max_bytes as usize;
439    loop {
440        let n = r.read(&mut scratch).map_err(|e| TardigradeError::Io { source: e })?;
441        if n == 0 {
442            return Ok(buf);
443        }
444        if buf.len().saturating_add(n) > max {
445            return Err(TardigradeError::SnapshotIntegrity(format!(
446                "entry exceeds maximum allowed size of {max_bytes} bytes",
447            )));
448        }
449        buf.extend_from_slice(&scratch[..n]);
450    }
451}

453fn hash_one(hasher: &mut Sha256, name: &std::ffi::OsStr, bytes: &[u8]) {
454    // Mix the path into the hash so reordered renames cannot collide
455    // with each other.
456    hasher.update(name.as_encoded_bytes());
457    hasher.update(b"\0");
458    hasher.update((bytes.len() as u64).to_le_bytes());
459    hasher.update(bytes);
460}

462fn hex_lower(bytes: &[u8]) -> String {
463    use std::fmt::Write;
464    let mut s = String::with_capacity(bytes.len() * 2);
465    for b in bytes {
466        let _ = write!(s, "{b:02x}");
467    }
468    s
