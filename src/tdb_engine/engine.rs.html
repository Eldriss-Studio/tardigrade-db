1//! Top-level engine: Facade over storage, retrieval, governance, and organization layers.
2//!
3//! Coordinates `BlockPool` + `BruteForceRetriever` + SLB + `VamanaIndex` + `TraceGraph` + WAL
4//! behind a unified `mem_write` / `mem_read` API.
5//!
6//! ## Retrieval Pipeline (Chain of Responsibility)
7//!
8//! ```text
9//! query → SLB (INT8, sub-5μs) → Vamana (graph ANN) → BruteForce (exact)
10//! ```
11//!
12//! Each stage returns results or passes to the next. Results are merged by `CellId`.
13

39/// Compute a mean-pooled key from a potentially per-token encoded key.
40///
41/// If the key is per-token encoded (has header), averages all token vectors.
42/// If it's already a plain vector, returns it unchanged.
43fn mean_pool_key(key: &[f32]) -> Vec<f32> {
44    if let Some((n, d, data)) = decode_per_token_keys(key) {
45        let mut mean = vec![0.0f32; d];
46        for i in 0..n {
47            for j in 0..d {
48                mean[j] += data[i * d + j];
49            }
50        }
51        let inv_n = 1.0 / n as f32;
52        for v in &mut mean {
53            *v *= inv_n;
54        }
55        mean
56    } else {
57        key.to_vec()
58    }
59}

65/// Adapter: wraps `VamanaIndex` (from `tdb-index`) to implement `Retriever` (from `tdb-retrieval`).
66///
67/// Bridges the interface gap: Vamana returns `(CellId, f32)` tuples with no owner info,
68/// while `Retriever` returns `RetrievalResult` with owner. The adapter sets owner to 0
69/// — the pipeline caller is responsible for final owner filtering.
70struct VamanaAdapter {
71    inner: VamanaIndex,
72}

74impl Retriever for VamanaAdapter {
75    fn query(
76        &mut self,
77        query_key: &[f32],
78        k: usize,
79        _owner_filter: Option<OwnerId>,
80    ) -> Vec<RetrievalResult> {
81        self.inner
82            .query(query_key, k)
83            .into_iter()
84            .map(|(cell_id, score)| RetrievalResult { cell_id, owner: 0, score })
85            .collect()
86    }
87
88    fn insert(&mut self, cell_id: CellId, _owner: OwnerId, key: &[f32]) {
89        self.inner.insert(cell_id, key);
90    }
91
92    fn len(&self) -> usize {
93        self.inner.len()
94    }

127/// Top-level `TardigradeDB` engine.
128///
129/// Facade pattern: single entry point coordinating storage, retrieval,
130/// governance, and organization (trace + WAL).
131///
132/// ## Retrieval Architecture
133///
134/// ```text
135/// query → SLB (cache, separate) → Pipeline [BruteForce, Vamana*] → merge
136/// ```
137///
138/// The SLB is a separate LRU cache (needs direct access for warming on reads).
139/// The [`RetrieverPipeline`] chains cold-path retrievers via the [`Retriever`] trait.
140/// Vamana is lazily activated when cell count crosses the threshold.
141/// Sentinel layer index for KV Pack retrieval key cells.
142const PACK_RETRIEVAL_LAYER: u16 = 0xFFFE;
143
144pub struct Engine {
145    pool: BlockPool,
146    /// Cold-path retrieval: `BruteForce` + Vamana (when active).
147    pipeline: RetrieverPipeline,
148    /// Hot-path cache: separate from pipeline for direct LRU warming on reads/writes.
149    slb: SemanticLookasideBuffer,
150    vamana: Option<VamanaIndex>,
151    trace: TraceGraph,
152    wal: Wal,
153    synaptic_store: SynapticStore,
154    governance: HashMap<CellId, CellGovernance>,
155    next_id: CellId,
156    dir: PathBuf,
157    vamana_threshold: usize,
158    /// Maps `PackId` to `Vec<CellId>` (retrieval key cell + layer payload cells).
159    pack_index: HashMap<PackId, Vec<CellId>>,
160    next_pack_id: PackId,
161    /// Key dimension (detected from first write, used for SLB/Vamana init).
162    key_dim: Option<usize>,
163}

181    fn open_with_options(
182        dir: &Path,
183        vamana_threshold: usize,
184        segment_size: Option<u64>,
185    ) -> Result<Self> {
186        let pool = match segment_size {
187            Some(size) => BlockPool::open_with_segment_size(dir, size)?,
188            None => BlockPool::open(dir)?,
189        };
190
191        // Open or create the SynapticStore (Repository for LoRA adapters).
192        let synaptic_store =
193            SynapticStore::open(dir).map_err(|e| TardigradeError::Io { source: e })?;
194
195        // Open or create the WAL for trace graph durability.
196        let wal = Wal::open(dir).map_err(|e| TardigradeError::WalRecovery(e.to_string()))?;
197
198        // Replay WAL to rebuild TraceGraph (Observer pattern recovery).
199        let mut trace = TraceGraph::new();
200        let wal_entries = wal.replay().map_err(|e| TardigradeError::WalRecovery(e.to_string()))?;
201        for entry in &wal_entries {
202            match entry {
203                WalEntry::AddEdge { src, dst, edge_type, timestamp } => {
204                    if let Some(et) = EdgeType::from_u8(*edge_type) {
205                        trace.add_edge(*src, *dst, et, *timestamp);
206                    }
207                }
208            }
209        }
210
211        // Rebuild derived state from persisted cells (Memento pattern).
212        // Pipeline: PerTokenRetriever (Top5Avg) → BruteForceRetriever (fallback).
213        let mut pipeline = RetrieverPipeline::new();
214        pipeline.add_stage(Box::new(
215            tdb_retrieval::per_token::PerTokenRetriever::with_scoring_mode(
216                tdb_retrieval::per_token::ScoringMode::Top5Avg,
217            ),
218        ));
219        pipeline.add_stage(Box::new(tdb_retrieval::attention::BruteForceRetriever::new()));
220
221        let mut governance = HashMap::new();
222        let mut next_id: CellId = 0;
223        let mut key_dim: Option<usize> = None;
224        let mut slb_entries: Vec<(CellId, OwnerId, Vec<f32>)> = Vec::new();
225
226        let cell_ids: Vec<CellId> = pool.iter_cell_ids().collect();
227
228        // Rebuild retrieval index from persisted cells.
229        // Vamana is added as a pipeline stage lazily when cell count crosses threshold.
230        for cell_id in &cell_ids {
231            let cell = pool.get(*cell_id)?;
232
233            // Index all cells for retrieval (including pack payload cells).
234            // This gives the scorer more candidates for discrimination.
235            let slb_key = mean_pool_key(&cell.key);
236            if key_dim.is_none() {
237                key_dim = Some(slb_key.len());
238            }
239
240            pipeline.insert(cell.id, cell.owner, &cell.key);
241            slb_entries.push((cell.id, cell.owner, slb_key));
242
243            let scorer = ImportanceScorer::new(cell.meta.importance);
244            let tier_sm = TierStateMachine::with_tier(cell.meta.tier);
245            governance.insert(cell.id, CellGovernance { scorer, tier_sm, days_since_update: 0.0 });
246
247            if cell.id >= next_id {
248                next_id = cell.id + 1;
249            }
250        }
251
252        let dim = key_dim.unwrap_or(128);
253        let mut slb = SemanticLookasideBuffer::new(DEFAULT_SLB_CAPACITY, dim);
254
255        // Populate SLB with mean-pooled keys (computed during rebuild above).
256        for (cell_id, owner, slb_key) in slb_entries {
257            slb.insert(cell_id, owner, &slb_key);
258        }
259
260        // Rebuild pack index from persisted cells.
261        // Pack cells store pack_id in token_span.0. PACK_RETRIEVAL_LAYER marks the key cell.
262        let mut pack_index: HashMap<PackId, Vec<CellId>> = HashMap::new();
263        let mut max_pack_id: PackId = 0;
264        for cell_id in &cell_ids {
265            if let Ok(cell) = pool.get(*cell_id) {
266                let stored_pack_id = cell.token_span.0;
267                if stored_pack_id > 0 || cell.layer == PACK_RETRIEVAL_LAYER {
268                    pack_index.entry(stored_pack_id).or_default().push(cell.id);
269                    if stored_pack_id >= max_pack_id {
270                        max_pack_id = stored_pack_id + 1;
271                    }
272                }
273            }
274        }
275        // Sort each pack's cells: retrieval key first (PACK_RETRIEVAL_LAYER), then by layer.
276        for cells in pack_index.values_mut() {
277            cells.sort_unstable();
278        }
279
280        Ok(Self {
281            pool,
282            pipeline,
283            slb,
284            vamana: None,
285            trace,
286            wal,
287            synaptic_store,
288            governance,
289            next_id,
290            dir: dir.to_path_buf(),
291            vamana_threshold,
292            pack_index,
293            next_pack_id: max_pack_id.max(1), // pack IDs start at 1 (0 = not a pack)
294            key_dim: if cell_ids.is_empty() { None } else { key_dim },
295        })
296    }

298    /// Write key/value vectors to the engine. Returns the assigned cell ID.
299    ///
300    /// `parent_cell_id`: optional causal parent for Trace graph edges.
301    pub fn mem_write(
302        &mut self,
303        owner: OwnerId,
304        layer: u16,
305        key: &[f32],
306        value: Vec<f32>,
307        salience: f32,
308        parent_cell_id: Option<CellId>,
309    ) -> Result<CellId> {
310        let id = self.next_id;
311        self.next_id += 1;
312
313        // Detect key dimension on first write (use mean-pooled dim for SLB).
314        if self.key_dim.is_none() {
315            let mean_dim = mean_pool_key(key).len();
316            self.key_dim = Some(mean_dim);
317            self.slb = SemanticLookasideBuffer::new(DEFAULT_SLB_CAPACITY, mean_dim);
318        }
319
320        let now_nanos = std::time::SystemTime::now()
321            .duration_since(std::time::UNIX_EPOCH)
322            .map_or(0, |d| d.as_nanos() as u64);
323
324        // Compute governance before persisting (Memento rebuild requirement).
325        let mut scorer = ImportanceScorer::new(salience);
326        scorer.on_update();
327        let mut tier_sm = TierStateMachine::new();
328        tier_sm.evaluate(scorer.importance());
329
330        let cell = MemoryCellBuilder::new(id, owner, layer, key.to_vec(), value)
331            .importance(scorer.importance())
332            .tier(tier_sm.current())
333            .created_at(now_nanos)
334            .updated_at(now_nanos)
335            .build();
336
337        // Persist to block pool.
338        self.pool.append(&cell)?;
339
340        // Index for retrieval.
341        // Pipeline gets the raw key (PerTokenRetriever decodes per-token encoding).
342        self.pipeline.insert(id, owner, key);
343
344        // SLB gets a mean-pooled summary (it needs fixed-dim vectors for INT8 scoring).
345        let slb_key = mean_pool_key(key);
346        self.slb.insert(id, owner, &slb_key);
347
348        self.governance.insert(id, CellGovernance { scorer, tier_sm, days_since_update: 0.0 });
349
350        // Trace graph: log causal edge via WAL (Observer pattern).
351        if let Some(parent_id) = parent_cell_id {
352            let wal_entry = WalEntry::AddEdge {
353                src: id,
354                dst: parent_id,
355                edge_type: EdgeType::CausedBy as u8,
356                timestamp: now_nanos,
357            };
358            self.wal.append(&wal_entry).map_err(|e| TardigradeError::WalRecovery(e.to_string()))?;
359            self.trace.add_edge(id, parent_id, EdgeType::CausedBy, now_nanos);
360        }
361
362        // Lazy Vamana activation: build when cell count crosses threshold.
363        if self.vamana.is_none() && self.pool.cell_count() >= self.vamana_threshold {
364            self.activate_vamana()?;
365        }
366
367        Ok(id)
368    }

370    /// Write multiple cells in a single batch with one fsync (Batch Command pattern).
371    ///
372    /// All cells are persisted to the block pool in one operation, then indexed
373    /// into the retrieval pipeline, SLB, and governance. Causal edges (if any)
374    /// are written to the WAL after the batch is persisted.
375    ///
376    /// Throughput: ~80us/cell amortized vs ~8ms/cell for individual writes.
377    pub fn mem_write_batch(&mut self, requests: &[WriteRequest]) -> Result<Vec<CellId>> {
378        if requests.is_empty() {
379            return Ok(Vec::new());
380        }
381
382        // Detect key dimension on first write (use mean-pooled dim for SLB).
383        if self.key_dim.is_none() {
384            let mean_dim = mean_pool_key(&requests[0].key).len();
385            self.key_dim = Some(mean_dim);
386            self.slb = SemanticLookasideBuffer::new(DEFAULT_SLB_CAPACITY, mean_dim);
387        }
388
389        let now_nanos = std::time::SystemTime::now()
390            .duration_since(std::time::UNIX_EPOCH)
391            .map_or(0, |d| d.as_nanos() as u64);
392
393        // Phase 1: Build all cells and compute governance (in memory, no I/O).
394        let mut cells = Vec::with_capacity(requests.len());
395        let mut gov_entries = Vec::with_capacity(requests.len());
396
397        for req in requests {
398            let id = self.next_id;
399            self.next_id += 1;
400
401            let mut scorer = ImportanceScorer::new(req.salience);
402            scorer.on_update();
403            let mut tier_sm = TierStateMachine::new();
404            tier_sm.evaluate(scorer.importance());
405
406            let cell = MemoryCellBuilder::new(
407                id,
408                req.owner,
409                req.layer,
410                req.key.clone(),
411                req.value.clone(),
412            )
413            .importance(scorer.importance())
414            .tier(tier_sm.current())
415            .created_at(now_nanos)
416            .updated_at(now_nanos)
417            .build();
418
419            gov_entries.push((
420                id,
421                req.owner,
422                req.key.clone(),
423                req.parent_cell_id,
424                CellGovernance { scorer, tier_sm, days_since_update: 0.0 },
425            ));
426            cells.push(cell);
427        }
428
429        // Phase 2: Persist all cells with single fsync.
430        let ids = self.pool.append_batch(&cells)?;
431
432        // Phase 3: Index all cells into retrieval + governance (in memory, fast).
433        for (id, owner, key, parent_cell_id, gov) in gov_entries {
434            self.pipeline.insert(id, owner, &key);
435            let slb_key = mean_pool_key(&key);
436            self.slb.insert(id, owner, &slb_key);
437            self.governance.insert(id, gov);
438
439            // Causal edges via WAL.
440            if let Some(parent_id) = parent_cell_id {
441                let wal_entry = WalEntry::AddEdge {
442                    src: id,
443                    dst: parent_id,
444                    edge_type: EdgeType::CausedBy as u8,
445                    timestamp: now_nanos,
446                };
447                self.wal
448                    .append(&wal_entry)
449                    .map_err(|e| TardigradeError::WalRecovery(e.to_string()))?;
450                self.trace.add_edge(id, parent_id, EdgeType::CausedBy, now_nanos);
451            }
452        }
453
454        // Lazy Vamana activation.
455        if self.vamana.is_none() && self.pool.cell_count() >= self.vamana_threshold {
456            self.activate_vamana()?;
457        }
458
459        Ok(ids)
460    }

462    /// Read the top-k most relevant cells for a query key.
463    ///
464    /// Two-phase retrieval: SLB (hot cache) then [`RetrieverPipeline`] (cold path).
465    /// Results are merged, deduplicated, scored with recency decay, and filtered by owner.
466    pub fn mem_read(
467        &mut self,
468        query_key: &[f32],
469        k: usize,
470        owner_filter: Option<OwnerId>,
471    ) -> Result<Vec<ReadResult>> {
472        if query_key.is_empty() {
473            return Ok(Vec::new());
474        }
475
476        let is_per_token_query = decode_per_token_keys(query_key).is_some();
477
478        // Phase 1: SLB hot cache (uses mean-pooled query for fixed-dim INT8 scoring).
479        let slb_query = mean_pool_key(query_key);
480        let mut candidates = Vec::new();
481
482        if is_per_token_query {
483            // Encoded per-token queries require max-sim scoring. SLB can fill gaps,
484            // but it must not suppress the cold per-token retriever.
485            let cold_results = self.pipeline.query(query_key, k * 2, owner_filter);
486            let mut seen: std::collections::HashSet<CellId> = std::collections::HashSet::new();
487            for r in cold_results {
488                if seen.insert(r.cell_id) {
489                    candidates.push(r);
490                }
491            }
492
493            if candidates.len() < k && !self.slb.is_empty() {
494                for r in self.slb.query(&slb_query, k * 2) {
495                    if seen.insert(r.cell_id) {
496                        candidates.push(r);
497                    }
498                }
499            }
500        } else {
501            candidates =
502                if self.slb.is_empty() { Vec::new() } else { self.slb.query(&slb_query, k * 2) };
503
504            // Phase 2: Cold-path pipeline (PerToken + BruteForce) — gets raw query.
505            if candidates.len() < k {
506                let cold_results = self.pipeline.query(query_key, k * 2, owner_filter);
507                // Deduplicate against SLB results.
508                let seen: std::collections::HashSet<CellId> =
509                    candidates.iter().map(|r| r.cell_id).collect();
510                for r in cold_results {
511                    if !seen.contains(&r.cell_id) {
512                        candidates.push(r);
513                    }
514                }
515            }
516        }
517
518        // Score adjustment + governance boost + cell retrieval.
519        candidates
520            .sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap_or(std::cmp::Ordering::Equal));
521
522        let mut results = Vec::with_capacity(k);
523        for rr in candidates {
524            let decay_factor = self
525                .governance
526                .get(&rr.cell_id)
527                .map_or(1.0, |g| recency_decay(g.days_since_update));
528
529            let adjusted_score = rr.score * decay_factor;
530
531            let tier =
532                self.governance.get(&rr.cell_id).map_or(Tier::Draft, |g| g.tier_sm.current());
533
534            match self.pool.get(rr.cell_id) {
535                Ok(cell) => {
536                    // Owner filter (pipeline may not filter internally for all stages).
537                    if let Some(filter_owner) = owner_filter
538                        && cell.owner != filter_owner
539                    {
540                        continue;
541                    }
542
543                    if let Some(gov) = self.governance.get_mut(&rr.cell_id) {
544                        gov.scorer.on_access();
545                        gov.tier_sm.evaluate(gov.scorer.importance());
546                    }
547                    // Warm the SLB with accessed cells for future hot-path hits.
548                    let slb_key = mean_pool_key(&cell.key);
549                    self.slb.insert(rr.cell_id, cell.owner, &slb_key);
550
551                    results.push(ReadResult { cell, score: adjusted_score, tier });
552                }
553                Err(TardigradeError::CellNotFound(_)) => continue,
554                Err(e) => return Err(e),
555            }
556
557            if results.len() >= k {
558                break;
559            }
560        }
561
562        results.sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap_or(std::cmp::Ordering::Equal));
563
564        Ok(results)
565    }

597    /// Simulate passage of time for governance decay.
598    pub fn advance_days(&mut self, days: f32) {
599        for gov in self.governance.values_mut() {
600            let old_whole = gov.days_since_update as u32;
601            gov.days_since_update += days;
602            let new_whole = gov.days_since_update as u32;
603            let elapsed = new_whole.saturating_sub(old_whole);
604            if elapsed > 0 {
605                gov.scorer.apply_daily_decay(elapsed);
606                gov.tier_sm.evaluate(gov.scorer.importance());
607            }
608        }
609    }

623    /// Store a complete multi-layer KV cache as a single atomic pack.
624    ///
625    /// All layers are persisted with a single fsync. The retrieval key is
626    /// indexed for search. Returns the assigned pack ID.
627    pub fn mem_write_pack(&mut self, pack: &KVPack) -> Result<PackId> {
628        let pack_id = self.next_pack_id;
629        self.next_pack_id += 1;
630
631        let now_nanos = std::time::SystemTime::now()
632            .duration_since(std::time::UNIX_EPOCH)
633            .map_or(0, |d| d.as_nanos() as u64);
634
635        // Detect key dimension from retrieval key.
636        let retrieval_key_pooled = mean_pool_key(&pack.retrieval_key);
637        if self.key_dim.is_none() {
638            let dim = retrieval_key_pooled.len();
639            self.key_dim = Some(dim);
640            self.slb = SemanticLookasideBuffer::new(DEFAULT_SLB_CAPACITY, dim);
641        }
642
643        // Build cells: one for retrieval key + one per layer payload.
644        let mut cells = Vec::with_capacity(1 + pack.layers.len());
645        let mut cell_ids = Vec::with_capacity(1 + pack.layers.len());
646
647        // Retrieval key cell (sentinel layer = PACK_RETRIEVAL_LAYER).
648        let retrieval_cell_id = self.next_id;
649        self.next_id += 1;
650
651        let mut scorer = ImportanceScorer::new(pack.salience);
652        scorer.on_update();
653        let mut tier_sm = TierStateMachine::new();
654        tier_sm.evaluate(scorer.importance());
655
656        // Store full per-token encoded key in the cell (for PerTokenRetriever scoring).
657        cells.push(
658            MemoryCellBuilder::new(
659                retrieval_cell_id,
660                pack.owner,
661                PACK_RETRIEVAL_LAYER,
662                pack.retrieval_key.clone(),
663                vec![], // no value for retrieval cell
664            )
665            .importance(scorer.importance())
666            .tier(tier_sm.current())
667            .token_span(pack_id, 0) // pack_id stored for cross-session rebuild
668            .created_at(now_nanos)
669            .updated_at(now_nanos)
670            .build(),
671        );
672        cell_ids.push(retrieval_cell_id);
673
674        // Layer payload cells.
675        for layer in &pack.layers {
676            let cell_id = self.next_id;
677            self.next_id += 1;
678
679            cells.push(
680                MemoryCellBuilder::new(
681                    cell_id,
682                    pack.owner,
683                    layer.layer_idx,
684                    pack.retrieval_key.clone(),
685                    layer.data.clone(),
686                )
687                .importance(scorer.importance())
688                .tier(tier_sm.current())
689                .token_span(pack_id, 0)
690                .created_at(now_nanos)
691                .updated_at(now_nanos)
692                .build(),
693            );
694            cell_ids.push(cell_id);
695        }
696
697        // Persist atomically (single fsync).
698        self.pool.append_batch(&cells)?;
699
700        // Index retrieval key for search — insert for EVERY cell in the pack
701        // (not just the retrieval cell) to give the scorer more candidates.
702        // This matches the old 28-cells-per-memory approach that achieves 8/10.
703        let slb_key = mean_pool_key(&pack.retrieval_key);
704        for &cid in &cell_ids {
705            self.pipeline.insert(cid, pack.owner, &pack.retrieval_key);
706            self.slb.insert(cid, pack.owner, &slb_key);
707        }
708
709        // Governance for the pack (tracked on retrieval cell).
710        self.governance
711            .insert(retrieval_cell_id, CellGovernance { scorer, tier_sm, days_since_update: 0.0 });
712
713        // Pack index.
714        self.pack_index.insert(pack_id, cell_ids);
715
716        Ok(pack_id)
717    }

719    /// Retrieve the top-k KV Packs matching a query key.
720    ///
721    /// Returns complete packs with all layer payloads reconstructed.
722    pub fn mem_read_pack(
723        &mut self,
724        query_key: &[f32],
725        k: usize,
726        owner_filter: Option<OwnerId>,
727    ) -> Result<Vec<PackReadResult>> {
728        // Search using the retrieval pipeline (same as mem_read).
729        let slb_query = mean_pool_key(query_key);
730        let mut candidates =
731            if self.slb.is_empty() { Vec::new() } else { self.slb.query(&slb_query, k * 2) };
732
733        if candidates.len() < k {
734            let cold = self.pipeline.query(query_key, k * 2, owner_filter);
735            let seen: std::collections::HashSet<CellId> =
736                candidates.iter().map(|r| r.cell_id).collect();
737            for r in cold {
738                if !seen.contains(&r.cell_id) {
739                    candidates.push(r);
740                }
741            }
742        }
743
744        candidates
745            .sort_by(|a, b| b.score.partial_cmp(&a.score).unwrap_or(std::cmp::Ordering::Equal));
746
747        let mut results = Vec::with_capacity(k);
748        let mut seen_packs: std::collections::HashSet<PackId> = std::collections::HashSet::new();
749
750        for rr in candidates {
751            // Find the pack that contains this cell (any cell in the pack).
752            let pack_entry =
753                self.pack_index.iter().find(|(_, cell_ids)| cell_ids.contains(&rr.cell_id));
754
755            let Some((&found_pack_id, cell_ids)) = pack_entry else {
756                continue;
757            };
758
759            // Deduplicate: skip if this pack already in results.
760            if !seen_packs.insert(found_pack_id) {
761                continue;
762            }
763
764            // Owner filter.
765            if owner_filter.is_some_and(|f| f != rr.owner) {
766                continue;
767            }
768
769            // Load layer payloads from pool.
770            let mut layers = Vec::new();
771            for &cid in &cell_ids[1..] {
772                match self.pool.get(cid) {
773                    Ok(cell) => {
774                        if cell.layer != PACK_RETRIEVAL_LAYER {
775                            layers.push(KVLayerPayload { layer_idx: cell.layer, data: cell.value });
776                        }
777                    }
778                    Err(TardigradeError::CellNotFound(_)) => {}
779                    Err(e) => return Err(e),
780                }
781            }
782
783            // Governance boost on access.
784            if let Some(gov) = self.governance.get_mut(&rr.cell_id) {
785                gov.scorer.on_access();
786                gov.tier_sm.evaluate(gov.scorer.importance());
787            }
788
789            let tier =
790                self.governance.get(&rr.cell_id).map_or(Tier::Draft, |g| g.tier_sm.current());
791
792            let decay_factor = self
793                .governance
794                .get(&rr.cell_id)
795                .map_or(1.0, |g| recency_decay(g.days_since_update));
796
797            layers.sort_by_key(|l| l.layer_idx);
798
799            results.push(PackReadResult {
800                pack: KVPack {
801                    id: found_pack_id,
802                    owner: rr.owner,
803                    retrieval_key: vec![], // not returned to save memory
804                    layers,
805                    salience: 0.0,
806                },
807                score: rr.score * decay_factor,
808                tier,
809            });
810
811            if results.len() >= k {
812                break;
813            }
814        }
815
816        Ok(results)
817    }

819    /// Number of KV Packs stored.
820    pub fn pack_count(&self) -> usize {
821        self.pack_index.len()
822    }

824    /// Get the importance score of a pack.
825    pub fn pack_importance(&self, pack_id: PackId) -> Option<f32> {
826        let cell_ids = self.pack_index.get(&pack_id)?;
827        let retrieval_cell_id = *cell_ids.first()?;
828        self.governance.get(&retrieval_cell_id).map(|g| g.scorer.importance())
829    }

831    /// Build and activate the Vamana index, adding it as a pipeline stage.
832    fn activate_vamana(&mut self) -> Result<()> {
833        let dim = self.key_dim.unwrap_or(128);
834        let mut vamana = VamanaIndex::new(dim, DEFAULT_VAMANA_MAX_DEGREE);
835
836        let cell_ids: Vec<CellId> = self.pool.iter_cell_ids().collect();
837        for cell_id in &cell_ids {
838            let cell = self.pool.get(*cell_id)?;
839            vamana.insert(cell.id, &cell.key);
840        }
841        vamana.build();
842
843        // Add Vamana as a pipeline stage via the adapter.
844        self.pipeline.add_stage(Box::new(VamanaAdapter { inner: vamana }));
845        self.vamana = Some(VamanaIndex::new(dim, DEFAULT_VAMANA_MAX_DEGREE)); // Marker only.
846        Ok(())
847    }
