1//! `BlockPoolCellSource` — Repository Adapter bridging `BlockPool` to
2//! [`CellSource`].
3//!
4//! The retrieval crate defines [`CellSource`] (one method:
5//! `get_encoded_key`). The storage crate defines `BlockPool` (a
6//! repository with `get(cell_id) -> Result<MemoryCell>`). This adapter
7//! is the thin glue that lets a lazy retriever (e.g.
8//! `PerTokenRetriever`) score cells stored in `BlockPool` without the
9//! retriever knowing anything about segments, fsync, or recovery.
10//!
11//! Held by borrowed reference (no `Arc`), constructed at query time
12//! by the engine.
13
14use tdb_core::CellId;
15use tdb_retrieval::cell_source::CellSource;
16use tdb_storage::block_pool::BlockPool;
17
18/// Repository Adapter: `&BlockPool` → `dyn CellSource`.
19#[derive(Debug)]
20pub struct BlockPoolCellSource<'a> {
21    pool: &'a BlockPool,
22}
23
24impl<'a> BlockPoolCellSource<'a> {
25    /// Wrap a borrowed `BlockPool` reference as a [`CellSource`].
26    pub fn new(pool: &'a BlockPool) -> Self {
27        Self { pool }
28    }
29}
30
31impl CellSource for BlockPoolCellSource<'_> {
32    fn get_encoded_key(&self, id: CellId) -> Option<Vec<f32>> {
33        self.pool.get(id).ok().map(|cell| cell.key)
34    }
35}
