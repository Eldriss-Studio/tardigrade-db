1//! Per-token importance weighting — Decorator pattern.
2//!
3//! Weight = `1 - cosine(token, corpus_mean)`. Distinctive tokens
4//! (orthogonal to mean) get high weight; common tokens get low weight.
5
6/// Compute the importance weight for a single token vector relative to the corpus mean.
7///
8/// Returns `1 - cosine(token, corpus_mean)`, clamped to `[0.0, 1.0]`.
9///
10/// - A token aligned with the mean (cosine ≈ 1) gets weight ≈ 0 — it is common/generic.
11/// - A token orthogonal to the mean (cosine ≈ 0) gets weight ≈ 1 — it is distinctive.
12/// - Zero-norm inputs return `1.0` (treat unknowns as fully weighted).
13pub fn token_weight(token: &[f32], corpus_mean: &[f32]) -> f32 {
14    let dot: f32 = token.iter().zip(corpus_mean).map(|(a, b)| a * b).sum();
15    let norm_t = token.iter().map(|x| x * x).sum::<f32>().sqrt();
16    let norm_m = corpus_mean.iter().map(|x| x * x).sum::<f32>().sqrt();
17    if norm_t < 1e-9 || norm_m < 1e-9 {
18        return 1.0;
19    }
20    (1.0 - dot / (norm_t * norm_m)).clamp(0.0, 1.0)
21}
22
23/// Return the token weight when `enabled`, or `1.0` when disabled.
24///
25/// The unit return allows callers to multiply scores by this value
26/// without branching in the hot path — the identity element `1.0` makes
27/// the multiplication a no-op when reweighting is off.
28pub fn weighted_or_unit(enabled: bool, token: &[f32], corpus_mean: &[f32]) -> f32 {
29    if enabled { token_weight(token, corpus_mean) } else { 1.0 }
30}

32#[cfg(test)]
33mod tests {
34    use super::*;
35
36    #[test]
37    fn weight_near_mean_is_low() {
38        let mean = [1.0_f32, 0.0, 0.0];
39        let token = [0.9_f32, 0.1, 0.0];
40        assert!(token_weight(&token, &mean) < 0.3);
41    }
42
43    #[test]
44    fn weight_orthogonal_is_high() {
45        let mean = [1.0_f32, 0.0, 0.0];
46        let token = [0.0_f32, 1.0, 0.0];
47        assert!(token_weight(&token, &mean) > 0.8);
48    }
49
50    #[test]
51    fn disabled_returns_one() {
52        assert!((weighted_or_unit(false, &[0.5, 0.5], &[1.0, 0.0]) - 1.0).abs() < 1e-6);
53    }
54
55    #[test]
56    fn weight_clamped_non_negative() {
57        let mean = [1.0_f32, 0.0];
58        let token = [1.0_f32, 0.0];
59        assert!(token_weight(&token, &mean) >= 0.0);
60    }
