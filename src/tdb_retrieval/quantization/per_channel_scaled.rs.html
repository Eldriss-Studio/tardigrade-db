1//! `PerChannelScaled<Q>` — Decorator that pre-scales tokens by a
2//! calibrated σ vector before delegating quantization to an inner
3//! strategy, and inverse-scales on dequantization.
4//!
5//! This is the `SmoothQuant` pattern applied to *storage* rather than
6//! inference. After dividing each dim by σ\[d\], every channel has
7//! roughly unit range — the activation-outlier channel no longer
8//! dominates its quantization group, and non-outlier dims retain
9//! their resolution.
10//!
11//! # Pattern: Decorator
12//!
13//! The wrapper exposes the same [`RetrievalQuantStrategy`] interface
14//! as its inner. New quantizers (`Q8Group32`, `NfFour`, future PQ
15//! impls) can be composed with per-channel scaling without changing
16//! either side — Open-Closed.
17
18use super::strategy::{QuantizedToken, RetrievalQuantStrategy};
19
20/// Decorator: pre-scale by σ, delegate to inner, inverse-scale on
21/// dequant.
22#[derive(Debug, Clone)]
23pub struct PerChannelScaled<Q: RetrievalQuantStrategy> {
24    inner: Q,
25    sigma: Vec<f32>,
26}
27
28impl<Q: RetrievalQuantStrategy> PerChannelScaled<Q> {
29    /// Wrap an inner strategy with a calibrated σ.
30    ///
31    /// `sigma.len()` must equal `inner.dim()`. Constructing with a
32    /// mismatched σ length yields a strategy that returns empty
33    /// quantized tokens for every input — fail-fast at score time.
34    pub fn new(inner: Q, sigma: Vec<f32>) -> Self {
35        Self { inner, sigma }
36    }
37
38    fn dims_match(&self) -> bool {
39        self.sigma.len() == self.inner.dim()
40    }
41}
42
43impl<Q: RetrievalQuantStrategy> RetrievalQuantStrategy for PerChannelScaled<Q> {
44    fn dim(&self) -> usize {
45        self.inner.dim()
46    }
47
48    fn quantize_token(&self, token: &[f32]) -> QuantizedToken {
49        if !self.dims_match() || token.len() != self.inner.dim() {
50            return QuantizedToken::empty();
51        }
52        let scaled: Vec<f32> =
53            token.iter().zip(self.sigma.iter()).map(|(value, sigma)| value / sigma).collect();
54        self.inner.quantize_token(&scaled)
55    }
56
57    fn dequantize_token(&self, quantized: &QuantizedToken) -> Vec<f32> {
58        if !self.dims_match() {
59            return Vec::new();
60        }
61        let mut out = self.inner.dequantize_token(quantized);
62        if out.len() != self.sigma.len() {
63            return Vec::new();
64        }
65        for (value, sigma) in out.iter_mut().zip(self.sigma.iter()) {
66            *value *= *sigma;
67        }
68        out
69    }
70}
