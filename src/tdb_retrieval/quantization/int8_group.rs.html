1//! `Int8Group32` — plain group-of-32 INT8 quantization for retrieval.
2//!
3//! Mirrors the `Q4_0` group structure used by `tdb-storage` for archival
4//! (one f32 scale per 32 values) but with 8-bit values instead of 4-bit.
5//! Suitable for retrieval scoring as the inner quantizer; pair with a
6//! [`PerChannelScaled`](super::per_channel_scaled::PerChannelScaled)
7//! decorator to neutralize activation outliers before encoding.
8//!
9//! # Why INT8 not Q4
10//!
11//! `Q4_0` has 16 levels per group (4-bit signed [-7, 7]). At Qwen3-style
12//! hidden-state magnitudes, the half-step is large enough to round
13//! non-outlier values to zero — destroying retrieval precision. `INT8`
14//! has 256 levels (signed [-127, 127]); the half-step is 16× smaller,
15//! and even on outlier-heavy groups the non-outlier dims remain
16//! resolvable. Production retrieval systems (`Faiss` SQ8, `Qdrant`,
17//! Sentence-Transformers) standardize on `INT8` as their precision
18//! floor for the same reason.
19
20use super::strategy::{QuantizedToken, RetrievalQuantStrategy};
21
22/// Group size for INT8 quantization. Matches the storage-layer Q4
23/// group size for alignment.
24pub const INT8_GROUP_SIZE: usize = 32;
25
26/// 8-bit signed integer range — half-width used when scaling
27/// `abs_max` into `[-INT8_LEVELS, INT8_LEVELS]` before clamping.
28/// `127` rather than `128` keeps quantization symmetric around
29/// zero.
30pub const INT8_LEVELS: f32 = 127.0;
31
32/// Floor applied to per-group `abs_max` to avoid divide-by-zero in
33/// all-zero groups. Mirrors
34/// [`PER_CHANNEL_SCALE_FLOOR`](super::calibrator::PER_CHANNEL_SCALE_FLOOR)
35/// but per-group rather than per-channel.
36pub const INT8_GROUP_SCALE_FLOOR: f32 = 1e-12;
37
38/// Plain INT8 group-of-32 quantizer. No outlier handling — see
39/// [`PerChannelScaled`](super::per_channel_scaled::PerChannelScaled)
40/// for the wrapper that pairs this with a calibrated σ to neutralize
41/// activation outliers.
42#[derive(Debug, Clone, Copy)]
43pub struct Int8Group32 {
44    dim: usize,
45}
46
47impl Int8Group32 {
48    /// Construct for a fixed per-token dimension.
49    pub fn new(dim: usize) -> Self {
50        Self { dim }
51    }
52}
53
54impl RetrievalQuantStrategy for Int8Group32 {
55    fn dim(&self) -> usize {
56        self.dim
57    }
58
59    fn quantize_token(&self, token: &[f32]) -> QuantizedToken {
60        if token.len() != self.dim {
61            return QuantizedToken::empty();
62        }
63        let num_groups = self.dim.div_ceil(INT8_GROUP_SIZE);
64        let mut packed_values = Vec::with_capacity(num_groups * INT8_GROUP_SIZE);
65        let mut group_scales = Vec::with_capacity(num_groups);
66        for chunk in token.chunks(INT8_GROUP_SIZE) {
67            let abs_max =
68                chunk.iter().map(|value| value.abs()).fold(INT8_GROUP_SCALE_FLOOR, f32::max);
69            let scale = abs_max / INT8_LEVELS;
70            group_scales.push(scale);
71            for value in chunk {
72                let scaled = (value / scale).round().clamp(-INT8_LEVELS, INT8_LEVELS);
73                packed_values.push(scaled as i8);
74            }
75            // Pad to group boundary so dequant length matches dim only
76            // after truncation. (Original length tracked via dim().)
77            let padding_needed = INT8_GROUP_SIZE.saturating_sub(chunk.len());
78            packed_values.extend(std::iter::repeat_n(0_i8, padding_needed));
79        }
80        QuantizedToken { values: packed_values, scales: group_scales, dim: self.dim }
81    }
82
83    fn dequantize_token(&self, quantized: &QuantizedToken) -> Vec<f32> {
84        if quantized.dim != self.dim {
85            return Vec::new();
86        }
87        let mut out = Vec::with_capacity(self.dim);
88        for (group_index, scale) in quantized.scales.iter().enumerate() {
89            let start = group_index * INT8_GROUP_SIZE;
90            let end = (start + INT8_GROUP_SIZE).min(quantized.values.len());
91            for value in &quantized.values[start..end] {
92                out.push(f32::from(*value) * scale);
93            }
94            if out.len() >= self.dim {
95                break;
96            }
97        }
98        out.truncate(self.dim);
99        out
100    }
101}

103#[cfg(test)]
104mod tests {
105    use super::*;
106
107    const TEST_DIM: usize = 64;
108
109    fn cosine(a: &[f32], b: &[f32]) -> f32 {
110        let dot: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
111        let norm_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
112        let norm_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
113        dot / (norm_a * norm_b).max(f32::MIN_POSITIVE)
114    }
115
116    const UNIFORM_RECALL_FLOOR: f32 = 0.999;
117
118    #[test]
119    fn uniform_token_round_trips_within_int8_envelope() {
120        let token: Vec<f32> = (0..TEST_DIM).map(|i| (i as f32 * 0.05).sin()).collect();
121        let q = Int8Group32::new(TEST_DIM);
122        let recovered = q.dequantize_token(&q.quantize_token(&token));
123        assert_eq!(recovered.len(), TEST_DIM);
124        let cos = cosine(&token, &recovered);
125        assert!(
126            cos >= UNIFORM_RECALL_FLOOR,
127            "uniform token cosine after INT8 round-trip {cos} below floor {UNIFORM_RECALL_FLOOR}",
128        );
129    }
