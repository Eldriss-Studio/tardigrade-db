1//! `RetrievalQuantStrategy` — Strategy trait for per-token retrieval
2//! quantization.
3//!
4//! Distinct from `tdb-storage`'s archival quantization (`QuantizeStrategy`
5//! over `Q4_0`). Retrieval quantization governs the *scoring* precision
6//! tier — INT8 (or finer) with per-channel scaling — and is independent
7//! of how cells are durably stored on disk.
8
9/// Opaque quantized representation of a single token vector.
10///
11/// `values` contains packed integers (the bit-width depends on the
12/// strategy that produced it). `scales` contains one per-group scale.
13/// `dim` is the original (pre-quantization) dimension, retained so
14/// dequantization can truncate any group-alignment padding.
15#[derive(Debug, Clone, PartialEq)]
16pub struct QuantizedToken {
17    pub values: Vec<i8>,
18    pub scales: Vec<f32>,
19    pub dim: usize,
20}
21
22impl QuantizedToken {
23    /// Construct an empty placeholder for the "wrong-dim input" or
24    /// "all-zero" cases. The retriever treats empty quantized tokens
25    /// as "skip."
26    pub fn empty() -> Self {
27        Self { values: Vec::new(), scales: Vec::new(), dim: 0 }
28    }
29
30    /// True when this token has no resolvable values.
31    pub fn is_empty(&self) -> bool {
32        self.dim == 0 || self.values.is_empty()
33    }
34}
35
36/// Strategy: how the retriever quantizes and dequantizes a single
37/// token vector for the scoring path.
38///
39/// Implementations must be deterministic — the same input always
40/// yields the same output — and must be `Send + Sync` because the
41/// engine holds them behind a shared reference and dispatches from
42/// concurrent query contexts.
43pub trait RetrievalQuantStrategy: Send + Sync {
44    /// Per-token dimension this strategy operates on.
45    fn dim(&self) -> usize;
46
47    /// Quantize one token vector.
48    ///
49    /// Returns [`QuantizedToken::empty()`] on input that doesn't match
50    /// `self.dim()` — callers treat this as "skip" rather than
51    /// failing the surrounding batch.
52    fn quantize_token(&self, token: &[f32]) -> QuantizedToken;
53
54    /// Dequantize one token back to f32.
55    ///
56    /// Returns an empty `Vec` if `quantized.dim` does not match
57    /// `self.dim()`. The recovered vector has length `self.dim()`.
58    fn dequantize_token(&self, quantized: &QuantizedToken) -> Vec<f32>;
59}
