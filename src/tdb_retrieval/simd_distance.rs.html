1//! SIMD-accelerated distance/dot-product functions.
2//!
3//! Strategy pattern: scalar fallback, with platform-specific optimizations.
4//! On aarch64 (Apple Silicon, ARM servers), uses NEON intrinsics.
5//! On `x86_64`, uses auto-vectorization-friendly loops.
6
7use crate::int8_quant::QuantizedInt8Vec;
8
9/// Dot product implementations for retrieval scoring.
10#[derive(Debug)]
11pub struct DotProduct;
12
13impl DotProduct {
14    /// FP32 dot product — reference implementation.
15    #[inline]
16    pub fn f32_dot(a: &[f32], b: &[f32]) -> f32 {
17        debug_assert_eq!(a.len(), b.len());
18        Self::f32_dot_scalar(a, b)
19    }
20
21    /// INT8 dot product with scale correction.
22    /// Returns an approximate FP32 result: `(scale_a * scale_b) * sum(a_q[i] * b_q[i])`.
23    #[inline]
24    pub fn int8_dot(a: &QuantizedInt8Vec, b: &QuantizedInt8Vec) -> f32 {
25        debug_assert_eq!(a.values.len(), b.values.len());
26        let raw = Self::int8_dot_raw(&a.values, &b.values);
27        raw as f32 * a.scale * b.scale
28    }
29
30    /// Raw INT8 dot product on slices — for `SoA` layout where data is contiguous.
31    #[inline]
32    pub fn int8_dot_raw_slice(a: &[i8], b: &[i8]) -> i32 {
33        debug_assert_eq!(a.len(), b.len());
34        Self::int8_dot_raw(a, b)
35    }
36
37    /// Scalar FP32 dot product. Written to auto-vectorize well:
38    /// simple loop, no branches, no dependencies between iterations.
39    #[inline]
40    fn f32_dot_scalar(a: &[f32], b: &[f32]) -> f32 {
41        a.iter().zip(b.iter()).map(|(x, y)| x * y).sum()
42    }
43
44    /// Raw INT8 dot product returning i32 accumulator.
45    /// On aarch64 with NEON, processes 16 elements per iteration.
46    #[cfg(target_arch = "aarch64")]
47    #[inline]
48    fn int8_dot_raw(a: &[i8], b: &[i8]) -> i32 {
49        use std::arch::aarch64::{
50            vaddvq_s32, vdupq_n_s32, vget_high_s8, vget_low_s8, vld1q_s8, vmull_s8, vpadalq_s16,
51        };
52
53        let len = a.len();
54        let chunks = len / 16;
55
56        // SAFETY: NEON intrinsics require `unsafe`. Pointer arithmetic is bounded by
57        // `chunks * 16 <= len`, so `a.as_ptr().add(offset)` and `b.as_ptr().add(offset)`
58        // always point within the slice. The `#[cfg(target_arch = "aarch64")]` gate
59        // ensures NEON is available.
60        let mut acc: i32 = unsafe {
61            let mut vacc = vdupq_n_s32(0);
62
63            for i in 0..chunks {
64                let offset = i * 16;
65                let va = vld1q_s8(a.as_ptr().add(offset));
66                let vb = vld1q_s8(b.as_ptr().add(offset));
67
68                // Multiply and pairwise widen: i8×i8 → i16, then accumulate to i32.
69                let low_a = vget_low_s8(va);
70                let high_a = vget_high_s8(va);
71                let low_b = vget_low_s8(vb);
72                let high_b = vget_high_s8(vb);
73
74                let prod_low = vmull_s8(low_a, low_b);
75                let prod_high = vmull_s8(high_a, high_b);
76
77                vacc = vpadalq_s16(vacc, prod_low);
78                vacc = vpadalq_s16(vacc, prod_high);
79            }
80
81            // Horizontal sum of 4×i32 lanes.
82            vaddvq_s32(vacc)
83        };
84
85        // Handle remaining elements.
86        let remainder_start = chunks * 16;
87        for i in remainder_start..len {
88            acc += a[i] as i32 * b[i] as i32;
89        }
90
91        acc
92    }

94    /// `x86_64`: `AVX2` when available, scalar fallback otherwise.
95    #[cfg(target_arch = "x86_64")]
96    #[inline]
97    fn int8_dot_raw(a: &[i8], b: &[i8]) -> i32 {
98        if is_x86_feature_detected!("avx2") {
99            // SAFETY: feature detection ensures AVX2 is available.
100            unsafe { Self::int8_dot_avx2(a, b) }
101        } else {
102            Self::int8_dot_scalar(a, b)
103        }
104    }
105
106    /// AVX2 INT8 dot product. Widens i8 to i16 then uses `vpmaddwd` for
107    /// pairwise multiply-add. Processes 32 elements per iteration.
108    ///
109    /// # Safety
110    ///
111    /// Caller must ensure AVX2 is available (checked via `is_x86_feature_detected!`).
112    /// Pointer arithmetic is bounded by `chunks * 32 <= len`.
113    /// `_mm256_loadu_si256` handles unaligned loads — the `cast` is safe for `loadu`.
114    #[cfg(target_arch = "x86_64")]
115    #[target_feature(enable = "avx2")]
116    #[expect(clippy::cast_ptr_alignment)]
117    unsafe fn int8_dot_avx2(a: &[i8], b: &[i8]) -> i32 {
118        use std::arch::x86_64::{
119            __m256i, _mm_add_epi32, _mm_cvtsi128_si32, _mm_srli_si128, _mm256_add_epi32,
120            _mm256_castsi256_si128, _mm256_cvtepi8_epi16, _mm256_extracti128_si256,
121            _mm256_loadu_si256, _mm256_madd_epi16, _mm256_setzero_si256,
122        };
123
124        let len = a.len();
125        let chunks = len / 32;
126        let mut acc = _mm256_setzero_si256();
127
128        for i in 0..chunks {
129            let off = i * 32;
130            // SAFETY: bounds checked above; loadu is unaligned-safe.
131            unsafe {
132                let va = _mm256_loadu_si256(a.as_ptr().add(off).cast::<__m256i>());
133                let vb = _mm256_loadu_si256(b.as_ptr().add(off).cast::<__m256i>());
134
135                let a_lo = _mm256_cvtepi8_epi16(_mm256_castsi256_si128(va));
136                let b_lo = _mm256_cvtepi8_epi16(_mm256_castsi256_si128(vb));
137                let a_hi = _mm256_cvtepi8_epi16(_mm256_extracti128_si256::<1>(va));
138                let b_hi = _mm256_cvtepi8_epi16(_mm256_extracti128_si256::<1>(vb));
139
140                acc = _mm256_add_epi32(acc, _mm256_madd_epi16(a_lo, b_lo));
141                acc = _mm256_add_epi32(acc, _mm256_madd_epi16(a_hi, b_hi));
142            }
143        }
144
145        let upper = _mm256_extracti128_si256::<1>(acc);
146        let lower = _mm256_castsi256_si128(acc);
147        let pair = _mm_add_epi32(lower, upper);
148        let half = _mm_add_epi32(pair, _mm_srli_si128::<8>(pair));
149        let quad = _mm_add_epi32(half, _mm_srli_si128::<4>(half));
150        let mut result = _mm_cvtsi128_si32(quad);
151
152        let tail = chunks * 32;
153        for i in tail..len {
154            result += a[i] as i32 * b[i] as i32;
155        }
156
157        result
158    }
159
160    /// Scalar fallback.
161    #[cfg(not(any(target_arch = "aarch64", target_arch = "x86_64")))]
162    #[inline]
163    fn int8_dot_raw(a: &[i8], b: &[i8]) -> i32 {
164        Self::int8_dot_scalar(a, b)
165    }
166
167    #[cfg(not(target_arch = "aarch64"))]
168    #[inline]
169    fn int8_dot_scalar(a: &[i8], b: &[i8]) -> i32 {
170        a.iter().zip(b.iter()).map(|(&x, &y)| x as i32 * y as i32).sum()
171    }

174#[cfg(test)]
175mod tests {
176    use super::*;
177
178    #[test]
179    fn test_f32_dot_basic() {
180        let a = vec![1.0, 2.0, 3.0];
181        let b = vec![4.0, 5.0, 6.0];
182        let result = DotProduct::f32_dot(&a, &b);
183        assert!((result - 32.0).abs() < f32::EPSILON);
184    }
185
186    #[test]
187    fn test_f32_dot_orthogonal() {
188        let a = vec![1.0, 0.0];
189        let b = vec![0.0, 1.0];
190        assert!((DotProduct::f32_dot(&a, &b)).abs() < f32::EPSILON);
191    }
192
193    #[test]
194    fn test_int8_dot_raw_basic() {
195        let a: Vec<i8> = vec![1, 2, 3, 4];
196        let b: Vec<i8> = vec![5, 6, 7, 8];
197        let result = DotProduct::int8_dot_raw(&a, &b);
198        assert_eq!(result, 5 + 2 * 6 + 3 * 7 + 4 * 8);
199    }
200
201    #[test]
202    fn test_int8_dot_raw_large_vector() {
203        let len = 128;
204        let a: Vec<i8> = (0..len).map(|i| (i % 10) as i8).collect();
205        let b: Vec<i8> = (0..len).map(|i| ((i + 3) % 10) as i8).collect();
206        let expected: i32 = a.iter().zip(b.iter()).map(|(&x, &y)| x as i32 * y as i32).sum();
207        let result = DotProduct::int8_dot_raw(&a, &b);
208        assert_eq!(result, expected);
209    }
