5/// A per-agent/user `LoRA` adapter pack for weight-like memory.
6///
7/// Unlike episodic KV memory (`MemoryCell`), synaptic entries encode
8/// stable preferences and patterns as low-rank weight deltas
9/// applied to the base model at runtime.
10#[derive(Debug, Clone, PartialEq)]
11pub struct SynapticBankEntry {
12    /// Unique identifier for this adapter set.
13    pub id: SynapticId,
14    /// The agent/user these adapters belong to.
15    pub owner: OwnerId,
16    /// `LoRA` matrix A (rank × `d_model`), stored in FP16.
17    pub lora_a: Vec<f16>,
18    /// `LoRA` matrix B (`d_model` × rank), stored in FP16.
19    pub lora_b: Vec<f16>,
20    /// Scaling factor applied to the low-rank delta.
21    pub scale: f16,
22    /// Rank of the `LoRA` decomposition.
23    pub rank: u32,
24    /// Model dimension (`d_model`).
25    pub d_model: u32,
26    /// Unix timestamp of last use.
27    pub last_used: u64,
28    /// Quality score from evaluation feedback.
29    pub quality: f32,
30}
31
32impl SynapticBankEntry {
33    /// Creates a new entry, validating that matrix dimensions match `rank` and `d_model`.
34    ///
35    /// # Panics
36    /// Panics if `lora_a.len() != rank * d_model` or `lora_b.len() != d_model * rank`.
37    pub fn new(
38        id: SynapticId,
39        owner: OwnerId,
40        lora_a: Vec<f16>,
41        lora_b: Vec<f16>,
42        scale: f16,
43        rank: u32,
44        d_model: u32,
45    ) -> Self {
46        let expected = (rank as usize) * (d_model as usize);
47        assert_eq!(
48            lora_a.len(),
49            expected,
50            "lora_a length {} does not match rank({}) × d_model({})",
51            lora_a.len(),
52            rank,
53            d_model
54        );
55        assert_eq!(
56            lora_b.len(),
57            expected,
58            "lora_b length {} does not match d_model({}) × rank({})",
59            lora_b.len(),
60            d_model,
61            rank
62        );
63        Self { id, owner, lora_a, lora_b, scale, rank, d_model, last_used: 0, quality: 0.0 }
64    }
65}

75            42,
76            vec![f16::from_f32(1.0); 8], // rank=2, d_model=4 → 8 elements

80            4,
