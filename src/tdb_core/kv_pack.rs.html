1//! KV Pack — a complete multi-layer KV cache stored as a single memory entity.
2//!
3//! A KV Pack captures the full `past_key_values` from one model inference pass.
4//! It contains per-layer K+V payloads for injection and a retrieval key (hidden
5//! state summary) for searching.
6//!
7//! ```text
8//! Model inference on "Sonia's wifi password is mango-cathedral-7"
9//!   │
10//!   ├── Layer 0:  K+V tensors ─┐
11//!   ├── Layer 1:  K+V tensors  │
12//!   ├── ...                    ├── KVPack (stored atomically, retrieved as unit)
13//!   ├── Layer 27: K+V tensors ─┘
14//!   │
15//!   └── Hidden states ──────────── Retrieval key (per-token encoded)
16//! ```
17//!
18//! Repository pattern: the engine treats a pack as a single entity for storage,
19//! retrieval, governance, and deletion — not as 28 scattered cells.
20
21use crate::{OwnerId, Tier};
22
23/// Unique identifier for a KV Pack.
24pub type PackId = u64;
25
26/// Per-layer K+V payload within a KV Pack.
27#[derive(Debug, Clone)]
28pub struct KVLayerPayload {
29    /// Transformer layer index.
30    pub layer_idx: u16,
31    /// Flattened `[K_flat | V_flat]` where each is `(seq_len * kv_heads * head_dim)` floats.
32    pub data: Vec<f32>,
33}
34
35/// A complete multi-layer KV cache captured from one inference pass.
36///
37/// Stored atomically (single fsync), retrieved as a unit, governed as one memory.
38/// Optionally carries the original fact text for durable text persistence
39/// alongside the tensor data.
40///
41/// # Examples
42///
43/// Construct a pack with the original fact text — the engine persists it
44/// in the durable text store alongside the tensor data:
45///
46/// ```rust
47/// use tdb_core::kv_pack::{KVLayerPayload, KVPack};
48///
49/// let pack = KVPack {
50///     id: 0, // assigned by the engine on write
51///     owner: 1,
52///     retrieval_key: vec![0.0f32; 128],
53///     layers: vec![KVLayerPayload {
54///         layer_idx: 0,
55///         data: vec![0.1f32; 64],
56///     }],
57///     salience: 80.0,
58///     text: Some("User prefers morning meetings".into()),
59/// };
60///
61/// assert_eq!(pack.text.as_deref(), Some("User prefers morning meetings"));
62/// ```
63///
64/// `text: None` is valid for packs that don't carry source text (e.g.,
65/// pre-Item-3 databases or pure-tensor workflows).
66#[derive(Debug, Clone)]
67pub struct KVPack {
68    /// Unique pack identifier (assigned by engine on write).
69    pub id: PackId,
70    /// Agent/user that owns this memory.
71    pub owner: OwnerId,
72    /// Per-token encoded hidden state summary for retrieval scoring.
73    ///
74    /// The encoded key uses the retrieval crate's 64-float Q4-safe header:
75    /// sentinel at index 0, metadata at indices 32 and 33, token data at 64+.
76    pub retrieval_key: Vec<f32>,
77    /// Per-layer K+V payloads for injection.
78    pub layers: Vec<KVLayerPayload>,
79    /// Initial importance hint (0-100).
80    pub salience: f32,
81    /// Original fact text, persisted in a durable append-only text store.
82    ///
83    /// `None` for packs created before text storage was added (backward compatible).
84    pub text: Option<String>,
85}
86
87/// Metadata about a stored pack, returned by retrieval.
88#[derive(Debug)]
89pub struct PackReadResult {
90    /// The complete KV Pack with all layer payloads.
91    pub pack: KVPack,
92    /// Retrieval score (attention-space similarity).
93    pub score: f32,
94    /// Current governance tier.
95    pub tier: Tier,
96}
